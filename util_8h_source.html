276  __attribute__((format(ZTH_ATTR_PRINTF, 1, 0))) inline std::string formatv(char const* fmt, va_list args) {
277  int const maxstack = (int)sizeof(void*) * 8;
278 
279  char sbuf[maxstack];
280  char* hbuf = NULL;
281  char* buf = sbuf;
282 
283  va_list args2;
284  va_copy(args2, args);
285  int c = vsnprintf(sbuf, maxstack, fmt, args);
286 
287  if(c >= maxstack) {
288  hbuf = (char*)malloc(c + 1);
289  if(unlikely(!hbuf))
290  c = 0;
291  else {
292  int c2 = vsprintf(hbuf, fmt, args2);
293  zth_assert(c2 <= c);
294  c = c2;
295  buf = hbuf;
296  }
297  }
298 
299  va_end(args2);
300 
301  std::string res(buf, (size_t)(c < 0 ? 0 : c));
302  if(hbuf)
303  free(hbuf);
304  return res;
305  }
