105 int execlp(char const* file, char const* arg, ... /*, nullptr */)
106 {
107  small_vector<char*,8> argv;
108  int res = 0;
109 
110  try {
111  va_list args;
112  va_start(args, arg);
113  while(true) {
114  char const* a = va_arg(args, char const*);
115  if(!a) {
116  argv.push_back(nullptr);
117  break;
118  } else {
119  char* argcopy = strdup(a);
120  if(!argcopy) {
121  res = ENOMEM;
122  break;
123  }
124  argv.push_back(argcopy);
125  }
126  }
127  va_end(args);
128  } catch(std::bad_alloc const&) {
129  res = ENOMEM;
130  } catch(...) {
131  res = EAGAIN;
132  }
133 
134  if(!res)
135  res = execvp(file, &argv[0]);
136 
137  for(size_t i = 0; i < argv.size(); i++)
138  if(argv[i])
139  free(argv[i]);
140 
141  return res;
142 }

164 int execvp(char const* UNUSED_PAR(file), char* const UNUSED_PAR(arg[]))
165 {
166 #if defined(ZTH_OS_WINDOWS) || defined(ZTH_OS_BAREMETAL)
167  return ENOSYS;
168 #else
169  perf_syscall("execvp()");
170  zth_dbg(thread, "[%s] execvp %s", thread_id_str().c_str(), file);
171 
172  pid_t pid = 0;
173  if((pid = fork()) == 0) {
174  // In child.
175  execve(file, arg, environ);
176  // If we get here, we could not create the process.
177  _exit(127);
178  // Unreachable.
179  return EAGAIN;
180  } else if(pid == -1) {
181  int res = errno;
182  zth_dbg(thread, "[%s] Could not fork(); %s", thread_id_str().c_str(),
183  err(res).c_str());
184  return res;
185  } else
186  return 0;
187 #endif
188 }

199 void sigchld_check()
200 {
201 #if !defined(ZTH_OS_WINDOWS) && !defined(ZTH_OS_BAREMETAL)
202 # ifdef ZTH_USE_VALGRIND
203  // The sigaction was disabled, so sigchld_cleanup will always be 0.
204  if(!RUNNING_ON_VALGRIND) // NOLINT(hicpp-no-assembler)
205 # endif
206  if(likely(sigchld_cleanup == 0))
207  return;
208 
209  Worker* w = Worker::instance();
210  char const* id_str = w ? w->id_str() : "?";
211 
212  while(true) {
213  sigchld_cleanup = 0;
214  int wstatus = 0;
215  pid_t pid = waitpid(-1, &wstatus, WNOHANG);
216 
217  if(pid == 0) {
218  // No (other) child exited.
219  return;
220  } else if(pid == -1) {
221  zth_dbg(worker, "[%s] waitpid() failed; %s", id_str, err(errno).c_str());
222  return;
223  } else {
224  zth_dbg(worker, "[%s] Child process %u terminated with exit code %d",
225  id_str, (unsigned int)pid, wstatus);
226  }
227  }
228 #endif
229 }
