121 int execvp(char const* UNUSED_PAR(file), char* const UNUSED_PAR(arg[])) {
122 #if defined(ZTH_OS_WINDOWS) || defined(ZTH_OS_BAREMETAL)
123  return ENOSYS;
124 #else
125  pid_t pid;
126  if((pid = vfork()) == 0) {
127  // In child.
128  execvp(file, arg);
129  // If we get here, we could not create the process.
130  _exit(127);
131  return EAGAIN;
132  } else if(pid == -1) {
133  int res = errno;
134  Worker* w = Worker::currentWorker();
135  char const* id_str = w ? w->id_str() : "?";
136  zth_dbg(worker, "[%s] Could not vfork(); %s", id_str, err(res).c_str());
137  return res;
138  } else
139  return 0;
140 #endif
141 }
