91  void flushEventBuffer()
92  {
93  zth_assert(Config::EnablePerfEvent);
94 
95  Fiber* f = fiber();
96  size_t spareRoom = eventBuffer().capacity() - eventBuffer().size();
97  if(unlikely(!f || f == m_worker.currentFiber() || spareRoom < 3)) {
98  // Do it right here right now.
99  processEventBuffer();
100  return;
101  }
102 
103  if(likely(spareRoom > 3)) {
104  // Wakeup and do the processing later on.
105  m_worker.resume(*f);
106  return;
107  }
108 
109  // Almost full, but enough room to signal that we are going to process the buffer.
110  Fiber* currentFiber_ = m_worker.currentFiber();
111  if(unlikely(!currentFiber_)) {
112  // Huh? Do it here anyway.
113  processEventBuffer();
114  return;
115  }
116 
117  // Record a 'context switch' to f...
118  PerfEvent<> e;
119  e.t = Timestamp::now();
120  e.fiber = f->id();
121  e.type = PerfEvent<>::Marker;
122  e.c_str = "Emerg flush";
123  eventBuffer().push_back(e);
124 
125  e.fiber = currentFiber_->id();
126  e.type = PerfEvent<>::FiberState;
127  e.fiberState = Fiber::Ready;
128  eventBuffer().push_back(e);
129 
130  e.fiber = f->id();
131  e.fiberState = Fiber::Running;
132  eventBuffer().push_back(e);
133 
134  zth_dbg(perf, "Emergency processEventBuffer()");
135  processEventBuffer();
136 
137  // ...and switch back.
138  e.t = Timestamp::now();
139  e.fiber = f->id();
140  e.fiberState = f->state();
141  eventBuffer().push_back(e);
142 
143  e.fiber = currentFiber_->id();
144  e.fiberState = currentFiber_->state();
145  eventBuffer().push_back(e);
146  }

176  int processEventBuffer()
177  {
178  int res = 0;
179  perf_eventBuffer_type& eb = eventBuffer();
180  size_t eb_size = eb.size();
181 
182  if(unlikely(!m_vcdd || !m_vcd)) {
183  res = EAGAIN;
184  goto done;
185  }
186 
187  for(size_t i = 0; i < eb_size; i++) {
188  PerfEvent<> const& e = eb[i];
189  TimeInterval t = e.t - startTime;
190  char const* s = nullptr;
191 
192  switch(e.type) {
193  case PerfEvent<>::FiberName:
194  if(e.str) {
195  for(char* p = e.str; *p; ++p) {
196  if(*p < 33 || *p > 126)
197  *p = '_';
198  }
199  string const& id = vcdId(e.fiber);
200  if(fprintf(m_vcd, "$var wire 1 %s \\#%" PRIu64 "_%s $end\n$var real 0 %s! \\#%" PRIu64 "_%s/log $end\n",
201  id.c_str(), e.fiber, e.str, id.c_str(), e.fiber, e.str) <= 0)
202  {
203  res = errno;
204  goto write_error;
205  }
206  } else {
207  string const& id = vcdId(e.fiber);
208  if(fprintf(m_vcd, "$var wire 1 %s %" PRIu64 "_Fiber $end\n$var real 0 %s! %" PRIu64 "_Fiber_log $end\n",
209  id.c_str(), e.fiber, id.c_str(), e.fiber) <= 0)
210  {
211  res = errno;
212  goto write_error;
213  }
214  }
215  break;
216 
217  case PerfEvent<>::FiberState:
218  {
219  char x = '-';
220  bool doCleanup = false;
221  switch(e.fiberState) {
222  case Fiber::New: x = 'L'; break;
223  case Fiber::Ready: x = '0'; break;
224  case Fiber::Running: x = '1'; break;
225  case Fiber::Waiting: x = 'W'; break;
226  case Fiber::Suspended: x = 'Z'; break;
227  case Fiber::Dead: x = 'X'; break;
228  default:
229  x = '-';
230  doCleanup = true;
231  }
232 
233  if(fprintf(m_vcdd, "#%" PRIu64 "%09ld\n%c%s\n", (uint64_t)t.ts().tv_sec, t.ts().tv_nsec, x, vcdId(e.fiber).c_str()) <= 0) {
234  res = errno;
235  goto write_error;
236  }
237 
238  if(doCleanup)
239  vcdIdRelease(e.fiber);
240 
241  break;
242  }
243 
244  case PerfEvent<>::Log:
245  {
246  if(true) { // NOLINT
247  s = e.c_str;
248  } else {
249  case PerfEvent<>::Marker:
250  s = e.str;
251  }
252  if(!s)
253  break;
254 
255  if(fprintf(m_vcdd, "#%" PRIu64 "%09ld\ns", (uint64_t)t.ts().tv_sec, t.ts().tv_nsec) <= 0) {
256  res = errno;
257  goto write_error;
258  }
259 
260  char const* chunkStart = s;
261  char const* chunkEnd = chunkStart;
262  int len = 0;
263  while(chunkEnd[1]) {
264  if(*chunkEnd < 33 || *chunkEnd > 126) {
265  if(fprintf(m_vcdd, "%.*s\\x%02hhx", len, chunkStart, (unsigned char)*chunkEnd) < 0) {
266  res = errno;
267  goto write_error;
268  }
269  chunkStart = chunkEnd = chunkEnd + 1;
270  len = 0;
271  } else {
272  chunkEnd++;
273  len++;
274  }
275  }
276 
277  if(fprintf(m_vcdd, "%s %s!\n", chunkStart, vcdId(e.fiber).c_str()) <= 0) {
278  res = errno;
279  goto write_error;
280  }
281  }
282 
283  default:
284  ; // ignore
285  }
286  }
287 
288  done:
289  for(size_t i = 0; i < eb_size; i++)
290  eb[i].release();
291  eb.clear();
292  return res;
293 
294  write_error:
295  fprintf(stderr, "Cannot write VCD file; %s\n", err(res).c_str());
296  if(!res)
297  res = EIO;
298  goto done;
299  }

301  int startVCD()
302  {
303  int res = 0;
304  char vcdFilename_[32];
305  char const* vcdFilename = snprintf(vcdFilename_, sizeof(vcdFilename_), "zth.%d.vcd", (int)getpid()) <= 0 ? vcdFilename_ : "zth.vcd";

307  if(!(m_vcd = fopen(vcdFilename, "w"))) {
308  res = errno;

311  }
312 
313 #ifdef ZTH_OS_WINDOWS
314  // tmpfile() on WIN32 is broken
315  m_vcdd = fopen("zth.vcdd", "w+");
316 #else

318 #endif

324  {

326  if(time(&now) != -1) {
327 #if defined(ZTH_OS_LINUX) || defined(ZTH_OS_MAC)

333 #endif

335  if(fprintf(m_vcd, "$date %s$end\n", strnow) < 0)
336  goto write_error;
337  }
338  }
339 
340  if(fprintf(m_vcd, "$version %s $end\n$timescale 1 ns $end\n$scope module top $end\n", banner()) < 0)
341  goto write_error;
342 
343  fprintf(stderr, "Using %s for perf VCD output\n", vcdFilename);
344  return 0;
345 
346  write_error:
347  fprintf(stderr, "Cannot write %s", vcdFilename);
348  res = EIO;
349  rollback:
350  if(m_vcdd) {
351  fclose(m_vcdd);
352  m_vcdd = nullptr;
353  }
354  if(m_vcd) {
355  fclose(m_vcd);
356  m_vcd = nullptr;
357  }
358  return res ? res : EIO;
359  }

361  int finalizeVCD()
362  {
363  if(!m_vcd || !m_vcdd)
364  return 0;
365 
366  int res = 0;
367  if(fprintf(m_vcd, "$upscope $end\n$enddefinitions $end\n") < 0)
368  goto write_error;
369 
370  rewind(m_vcdd);
371 
372  {
373  char buf[1024];
374  size_t cnt = 0;
375  while((cnt = fread(buf, 1, sizeof(buf), m_vcdd)) > 0)
376  if(fwrite(buf, cnt, 1, m_vcd) != 1)
377  goto write_error;
378  }
379 
380  if(ferror(m_vcdd))
381  goto read_error;
382 
383  zth_assert(feof(m_vcdd));
384  zth_assert(!ferror(m_vcd));
385 
386  fclose(m_vcdd);
387  m_vcdd = nullptr;
388  fclose(m_vcd);
389  m_vcd = nullptr;
390  return 0;
391 
392  read_error:
393  fprintf(stderr, "Cannot read temporary VCD data file\n");
394  return res ? res : EIO;
395  write_error:
396  fprintf(stderr, "Cannot write VCD file\n");
397  return res ? res : EIO;
398  }

400  string const& vcdId(uint64_t fiber)
401  {
402  std::pair<decltype(m_vcdIds.begin()),bool> i = m_vcdIds.insert(std::make_pair(fiber, string()));
403  if(likely(!i.second))
404  // Already in the map. Return the value.
405  return i.first->second;
406 
407  // Just added with empty string. Generate a proper VCD identifier.
408  // Identifier is a string of ASCII 34-126. Use 33 (!) as special character.
409  // Do a radix-93 convert.
410  uint64_t id = fiber;
411  char buf[16]; // the string cannot be longer than 10 chars, as 93^10 > 2^64.
412  char* s = &buf[sizeof(buf) - 1];
413  *s = 0;
414 
415  do {
416  *--s = (char)(id % 93 + 34);
417  id /= 93;
418  } while(id);
419 
420  return i.first->second = s;
421  }

423  void vcdIdRelease(uint64_t fiber)
424  {
425  decltype(m_vcdIds.begin()) it = m_vcdIds.find(fiber);
426  if(it != m_vcdIds.end())
427  m_vcdIds.erase(it);
428  }

469 Backtrace::Backtrace(size_t UNUSED_PAR(skip), size_t UNUSED_PAR(maxDepth))
470  : m_t0(Timestamp::now())
471 {
472  Worker* worker = Worker::instance();
473  m_fiber = worker ? worker->currentFiber() : nullptr;
474  m_fiberId = m_fiber ? m_fiber->id() : 0;
475  m_truncated = true;
476 
477 #ifdef ZTH_HAVE_LIBUNWIND
478  unw_context_t uc;
479 
480  if(unw_getcontext(&uc))
481  return;
482 
483  unw_cursor_t cursor;
484  unw_init_local(&cursor, &uc);
485  size_t depth = 0;
486  for(size_t i = 0; i < skip && unw_step(&cursor) > 0; i++);
487 
488  unw_proc_info_t pip;
489 
490  while (unw_step(&cursor) > 0 && depth < maxDepth) {
491  if(depth == 0) {
492  unw_word_t sp = 0;
493  unw_get_reg(&cursor, UNW_REG_SP, &sp);
494  }
495 
496  unw_word_t ip = 0;
497  unw_get_reg(&cursor, UNW_REG_IP, &ip);
498  m_bt.push_back(reinterpret_cast<void*>(ip));
499 
500  if(unw_get_proc_info(&cursor, &pip) == 0 && pip.start_ip == reinterpret_cast<unw_word_t>(&context_entry))
501  // Stop here, as we might get segfaults when passing the context_entry functions.
502  break;
503  }
504 
505  m_truncated = depth == maxDepth;
506 #elif !defined(ZTH_OS_WINDOWS) && !defined(ZTH_OS_BAREMETAL)
507  m_bt.resize(maxDepth);
508  m_bt.resize((size_t)backtrace(&m_bt[0], (int)maxDepth));
509  m_truncated = m_bt.size() == maxDepth;
510 #endif
511 
512  m_t1 = Timestamp::now();
513 }

603 void Backtrace::printDelta(Backtrace const& other, int color) const
604 {
605  // Make sure other was first.
606  if(other.t0() > t0()) {
607  other.printDelta(*this, color);
608  return;
609  }
610 
611  TimeInterval dt = t0() - other.t1();
612 
613  if(other.fiberId() != fiberId() ||
614  other.truncated() || truncated())
615  {
616  log_color(color, "%sExecuted from:\n", color >= 0 ? ZTH_DBG_PREFIX : "");
617  other.print(color);
618  log_color(color, "%sto:\n", color >= 0 ? ZTH_DBG_PREFIX : "");
619  print(color);
620  log_color(color, "%stook %s\n", color >= 0 ? ZTH_DBG_PREFIX : "", dt.str().c_str());
621  return;
622  }
623 
624  // Find common base
625  ssize_t common = 0;
626  while(bt()[bt().size() - 1 - (size_t)common] == other.bt()[other.bt().size() - 1 - (size_t)common])
627  common++;
628 
629  log_color(color, "%sExecution from fiber %p #%" PRIu64 ":\n", color >= 0 ? ZTH_DBG_PREFIX : "", m_fiber, m_fiberId);
630  other.printPartial(0, -common - 1, color);
631  log_color(color, "%sto:\n", color >= 0 ? ZTH_DBG_PREFIX : "");
632  printPartial(0, -common - 1, color);
633  if(common > 0) {
634  log_color(color, "%shaving in common:\n", color >= 0 ? ZTH_DBG_PREFIX : "");
635  other.printPartial(other.bt().size() - (size_t)common, color);
636  }
637  log_color(color, "%stook %s\n", color >= 0 ? ZTH_DBG_PREFIX : "", dt.str().c_str());
638 }
