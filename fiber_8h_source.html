163  int run(Fiber& from, Timestamp now = Timestamp::now())
164  {
165  int res = 0;
166 
167  again:
168  switch(state())
169  {
170  case New:
171  // First call, do implicit init.
172  if((res = init(now)))
173  return res;
174 
175  if(hookNew)
176  hookNew(*this);
177  goto again;
178 
179  case Ready:
180  {
181  zth_assert(&from != this);
182 
183  // Update administration of the current fiber.
184  TimeInterval dt = now - from.m_startRun;
185  from.m_totalTime += dt;
186 
187  if(from.state() == Running)
188  from.setState(Ready, now);
189 
190  if(unlikely(zth_config(CheckTimesliceOverrun) && from.m_dtMax < dt)) {
191  perf_mark("timeslice overrun reported");
192  from.m_dtMax = dt;
193  log_color(Config::Print_perf, ZTH_DBG_PREFIX "Long timeslice by %s of %s\n", from.id_str(), dt.str().c_str());
194  Backtrace().print(Config::Print_perf);
195  now = Timestamp::now();
196  }
197 
198  // Hand over to this.
199  m_startRun = now;
200  setState(Running, now);
201  m_stateEnd = now + m_timeslice;
202 
203  zth_dbg(fiber, "Switch from %s to %s after %s", from.id_str(), id_str(), dt.str().c_str());
204  context_switch(from.context(), context());
205 
206  // Ok, got back.
207  // Warning! This fiber might already be dead and destructed at this point!
208  // Only return here!
209  return 0;
210  }
211 
212  case Running:
213  zth_assert(&from == this);
214  // No switch required.
215  return EAGAIN;
216 
217  case Dead:
218  // Can't run dead fibers.
219  return EPERM;
220 
221  default:
222  zth_assert(false);
223  return EINVAL;
224  }
225  }

241  void nap(Timestamp const& sleepUntil = Timestamp::null()) noexcept
242  {
243  switch(state()) {
244  case New:
245  // Postpone actual sleep
246  zth_dbg(fiber, "[%s] Sleep upon startup", id_str());
247  m_stateNext = Waiting;
248  break;
249  case Ready:
250  case Running:
251  if(sleepUntil.isNull())
252  zth_dbg(fiber, "[%s] Sleep", id_str());
253  else
254  zth_dbg(fiber, "[%s] Sleep for %s", id_str(), (sleepUntil - Timestamp::now()).str().c_str());
255  setState(Waiting);
256  m_stateNext = Ready;
257  break;
258  case Suspended:
259  zth_dbg(fiber, "[%s] Schedule sleep after resume", id_str());
260  m_stateNext = Waiting;
261  break;
262  case Waiting:
263  default:
264  return;
265  }
266 
267  m_stateEnd = sleepUntil;
268  }

270  void wakeup() noexcept
271  {
272  if(likely(state() == Waiting)) {
273  setState(m_stateNext);
274  switch(state()) {
275  case Suspended:
