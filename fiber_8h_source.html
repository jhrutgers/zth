117  int init(Timestamp const& now = Timestamp::now()) {
118  if(state() != New)
119  return EPERM;
120 
121  zth_assert(!m_context);
122 
123  zth_dbg(fiber, "[%s] Init", id_str());
124  int res = context_create(m_context, m_contextAttr);
125  if(res) {
126  // Oops.
127  kill();
128  return res;
129  }

138  int run(Fiber& from, Timestamp now = Timestamp::now()) {
139  int res = 0;
140 
141  again:
142  switch(state())
143  {
144  case New:
145  // First call, do implicit init.
146  if((res = init(now)))
147  return res;
148  goto again;
149 
150  case Ready:
151  {
152  // Update administration of the current fiber.
153  TimeInterval dt = now - from.m_startRun;
154  from.m_totalTime += dt;
155 
156  if(from.state() == Running)
157  from.setState(Ready, now);
158 
159  if(unlikely(zth_config(CheckTimesliceOverrun) && from.m_dtMax < dt)) {
160  perf_mark("timeslice overrun reported");
161  from.m_dtMax = dt;
162  log_color(Config::Print_perf, ZTH_DBG_PREFIX "Long timeslice by %s of %s\n", from.id_str(), dt.str().c_str());
163  Backtrace().print(Config::Print_perf);
164  now = Timestamp::now();
165  }
166 
167  // Hand over to this.
168  m_startRun = now;
169  setState(Running, now);
170  m_stateEnd = now + m_timeslice;
171 
172  zth_dbg(fiber, "Switch from %s to %s after %s", from.id_str(), id_str(), dt.str().c_str());
173  context_switch(from.context(), context());
174 
175  // Ok, got back.
176  // Warning! This fiber might already be dead and destructed at this point!
177  // Only return here!
178  return 0;
179  }
180 
181  case Running:
182  zth_assert(&from == this);
183  // No switch required.
184  return EAGAIN;
185 
186  case Dead:
187  // Can't run dead fibers.
188  return EPERM;
189 
190  default:
191  zth_assert(false);
192  return EINVAL;
193  }
194  }

208  void nap(Timestamp const& sleepUntil = Timestamp::null()) {
209  switch(state()) {
210  case New:
211  // Postpone actual sleep
212  zth_dbg(fiber, "[%s] Sleep upon startup", id_str());
213  m_stateNext = Waiting;
214  break;
215  case Ready:
216  case Running:
217  if(sleepUntil.isNull())
218  zth_dbg(fiber, "[%s] Sleep", id_str());
219  else
220  zth_dbg(fiber, "[%s] Sleep for %s", id_str(), (sleepUntil - Timestamp::now()).str().c_str());
221  setState(Waiting);
222  m_stateNext = Ready;
223  break;
224  case Suspended:
225  zth_dbg(fiber, "[%s] Schedule sleep after resume", id_str());
226  m_stateNext = Waiting;
227  break;
228  case Waiting:
229  default:
230  return;
231  }
232 
233  m_stateEnd = sleepUntil;
234  }

236  void wakeup() {
237  if(likely(state() == Waiting)) {
238  setState(m_stateNext);
239  switch(state()) {
240  case Suspended:
241  zth_dbg(fiber, "[%s] Suspend after wakeup", id_str());
242  m_stateNext = Ready;
243  break;
244  default:
245  zth_dbg(fiber, "[%s] Wakeup", id_str());
246  }
247  } else if(state() == Suspended && m_stateNext == Waiting) {
248  zth_dbg(fiber, "[%s] Set wakeup after suspend", id_str());
249  m_stateNext = Ready;
250  }
251  }

253  void suspend() {
254  switch(state()) {
255  case New:
256  m_stateNext = New;
257  break;
258  case Running:
259  case Ready:
260  m_stateNext = Ready;
261  break;
262  case Waiting:
263  m_stateNext = Waiting;
264  case Suspended:
265  default:
266  // Ignore.
267  return;
268  }
269 
270  zth_dbg(fiber, "[%s] Suspend", id_str());
271  setState(Suspended);
272  }

276  return;

278  zth_dbg(fiber, "[%s] Resume", id_str());
279  setState(m_stateNext);
