163  void add(TimeInterval const& t) {
164  if(t.m_negative == m_negative) {
165  // Check for overflow.
166  if(isInfinite() || t.isInfinite() || std::numeric_limits<time_t>::max() - m_t.tv_sec - 1 <= t.m_t.tv_sec) {
167  m_t.tv_sec = std::numeric_limits<time_t>::max(); // infinite
168  } else {
169  // Add in same sign direction.
170  m_t.tv_sec += t.m_t.tv_sec;
171  m_t.tv_nsec += t.m_t.tv_nsec;
172  if(m_t.tv_nsec > BILLION) {
173  m_t.tv_nsec -= BILLION;
174  m_t.tv_sec++;
175  }
176  }
177  } else {
178  if(t.isAbsBiggerThan(*this)) {
179  // Add in other sign direction with sign flip.
180  m_t.tv_sec = t.m_t.tv_sec - m_t.tv_sec;
181  m_t.tv_nsec = t.m_t.tv_nsec - m_t.tv_nsec;
182  m_negative = !m_negative;
183  } else {
184  // Add in other sign direction (but without sign flip).
185  m_t.tv_sec -= t.m_t.tv_sec;
186  m_t.tv_nsec -= t.m_t.tv_nsec;
187  }
188  if(m_t.tv_nsec < 0) {
189  m_t.tv_nsec += BILLION;
190  m_t.tv_sec--;
191  }
192  }
193  }
