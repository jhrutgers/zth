147  void add(TimeInterval const& t) {
148  if(t.m_negative == m_negative) {
149  // Check for overflow.
150  if(isInfinite() || t.isInfinite() || std::numeric_limits<time_t>::max() - m_t.tv_sec - 1 <= t.m_t.tv_sec) {
151  m_t.tv_sec = std::numeric_limits<time_t>::max(); // infinite
152  } else {
153  // Add in same sign direction.
154  m_t.tv_sec += t.m_t.tv_sec;
155  m_t.tv_nsec += t.m_t.tv_nsec;
156  if(m_t.tv_nsec > BILLION) {
157  m_t.tv_nsec -= BILLION;
158  m_t.tv_sec++;
159  }
160  }
161  } else {
162  if(t.isAbsBiggerThan(*this)) {
163  // Add in other sign direction with sign flip.
164  m_t.tv_sec = t.m_t.tv_sec - m_t.tv_sec;
165  m_t.tv_nsec = t.m_t.tv_nsec - m_t.tv_nsec;
166  m_negative = !m_negative;
167  } else {
168  // Add in other sign direction (but without sign flip).
169  m_t.tv_sec -= t.m_t.tv_sec;
170  m_t.tv_nsec -= t.m_t.tv_nsec;
171  }
172  if(m_t.tv_nsec < 0) {
173  m_t.tv_nsec += BILLION;
174  m_t.tv_sec--;
175  }
176  }
177  }
