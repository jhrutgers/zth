187  constexpr14 void add(TimeInterval const& t) noexcept
188  {
189  if(t.m_negative == m_negative) {
190  // Check for overflow.
191  if(isInfinite() || t.isInfinite() || std::numeric_limits<time_t>::max() - m_t.tv_sec - 1 <= t.m_t.tv_sec) {
192  m_t.tv_sec = std::numeric_limits<time_t>::max(); // infinite
193  } else {
194  // Add in same sign direction.
195  m_t.tv_sec += t.m_t.tv_sec;
196  m_t.tv_nsec += t.m_t.tv_nsec;
197  if(m_t.tv_nsec > BILLION) {
198  m_t.tv_nsec -= BILLION;
199  m_t.tv_sec++;
200  }
201  }
202  } else {
203  if(t.isAbsBiggerThan(*this)) {
204  // Add in other sign direction with sign flip.
205  m_t.tv_sec = t.m_t.tv_sec - m_t.tv_sec;
206  m_t.tv_nsec = t.m_t.tv_nsec - m_t.tv_nsec;
207  m_negative = !m_negative;
208  } else {
209  // Add in other sign direction (but without sign flip).
210  m_t.tv_sec -= t.m_t.tv_sec;
211  m_t.tv_nsec -= t.m_t.tv_nsec;
212  }
213  if(m_t.tv_nsec < 0) {
214  m_t.tv_nsec += BILLION;
215  m_t.tv_sec--;
216  }
217  }
218  }
