161  static int stackGuardRegion() noexcept
162  {
163  if(!Config::EnableStackGuard || Config::EnableThreads)
164  return -1;
165 
166  static int region = 0;
167 
168  if(likely(region))
169  return region;
170 
171  region = reg_mpu_type().field.dregion - 1;
172  reg_mpu_rnr rnr;
173  reg_mpu_rasr rasr;
174 
175  // Do not try to use region 0, leave at least one unused for other application
176  // purposes.
177  for(; region > 0; --region) {
178  rnr.field.region = region;
179  rnr.write();
180 
181  rasr.read();
182  if(!rasr.field.enable)
183  // This one is free.
184  break;
185  }
186 
187  if(region > 0) {
188  reg_mpu_rbar rbar;
189  // Initialize at the top of the address space, so it is effectively unused.
190  rbar.field.addr = REG_MPU_RBAR_ADDR_UNUSED;
191  rbar.write();
192 
193  rasr.field.size = ZTH_ARM_STACK_GUARD_BITS - 1;
194  rasr.field.xn = 0;
195  rasr.field.ap = 0; // no access
196 
197  // Internal RAM should be normal, shareable, write-through.
198  rasr.field.tex = 0;
199  rasr.field.c = 1;
200  rasr.field.b = 0;
201  rasr.field.s = 1;
202 
203  rasr.field.enable = 1;
204  rasr.write();
205 
206  reg_mpu_ctrl ctrl;
207  if(!ctrl.field.enable) {
208  // MPU was not enabled at all.
209  ctrl.field.privdefena = 1; // Assume we are currently privileged.
210  ctrl.field.enable = 1;
211  ctrl.write();
212  __isb();
213  }
214  zth_dbg(context, "Using MPU region %d as stack guard", region);
215  } else {
216  region = -1;
217  zth_dbg(context, "Cannot find free MPU region for stack guard");
218  }
219 
220  return region;
221  }

223  static void stackGuard(void* guard) noexcept
224  {
225  if(!Config::EnableStackGuard || Config::EnableThreads)
226  return;
227 
228  int const region = stackGuardRegion();
229  if(unlikely(region < 0))
230  return;
231 
232  // Must be aligned to guard size
233  zth_assert(((uintptr_t)guard & (ZTH_ARM_STACK_GUARD_SIZE - 1)) == 0);
234 
235  reg_mpu_rbar rbar(0);
236 
237  if(likely(guard)) {
238  rbar.field.addr = (uintptr_t)guard >> 5;
239  zth_dbg(context, "Set MPU stack guard to %p (sp=%p)", guard, sp());
240  } else {
241  // Initialize at the top of the address space, so it is effectively unused.
242  rbar.field.addr = REG_MPU_RBAR_ADDR_UNUSED;
243  zth_dbg(context, "Disabled MPU stack guard");
244  }
245 
246  rbar.field.valid = 1;
247  rbar.field.region = region;
248  rbar.write();
249 
250  if(Config::Debug) {
251  // Only required if the settings should be in effect immediately.
252  // However, it takes some time, which may not be worth it in a release
253  // build. If skipped, it will take affect, but only after a few
254  // instructions, which are already in the pipeline.
255  __dsb();
256  __isb();
257  }

283  static size_t get_lr_offset(jmp_buf const& env) noexcept
284  {
285  static size_t lr_offset = 0;
286 
287  if(unlikely(lr_offset == 0)) {
288  // Not initialized yet.
289 
290  // env is filled with several registers. The number of
291  // registers depends on the ARM architecture and Thumb mode,
292  // and compile settings for newlib. We have to change the sp
293  // and lr registers in this env. These are always the last two
294  // registers in the list, and are always non-zero. So, find
295  // the last non-zero register to determine the lr offset.
296 
297  lr_offset = sizeof(env) / sizeof(env[0]) - 1;
298  for(; lr_offset > 1 && !env[lr_offset]; lr_offset--)
299  ;
300 
301  if(lr_offset <= 1) {
302  // Should not be possible.
303  zth_abort("Invalid lr offset");
304  }
305  }
306 
307  return lr_offset;
308  }

369 # ifdef ZTH_ARM_USE_PSP
370  inline __attribute__((always_inline)) void
371  context_prepare_jmp(Impl& to, jmp_buf& env) noexcept
372  {
373  // Use the PSP for fibers and the MSP for the worker.
374  // So, if there is no stack in the Context, assume it is a worker, running
375  // from MSP. Otherwise it is a fiber using PSP. Both are executed privileged.
376  if(unlikely(!impl().stackUsable() || !to.stackUsable())) {
377  unsigned int control;
378  asm("mrs %0, control\n" : "=r"(control));
379 
380  if(to.stackUsable())
381  control |= 2u; // set SPSEL to PSP
382  else
383  control &= ~2u; // set SPSEL to MSR
384 
385  // As the SP changes, do not return from this function,
386  // but actually do the longjmp.
387  asm volatile(
388  "msr control, %1\n"
389  "isb\n"
390  "mov r0, %0\n"
391  "movs r1, #1\n"
392  "b longjmp\n"
393  :
394  : "r"(env), "r"(control)
395  : "r0", "r1", "memory");
396  }
397  }
398 # endif
399 # endif // ZTH_CONTEXT_SJLJ

404  __attribute__((naked, noinline)) static void*
405  stack_switch_msp(void* UNUSED_PAR(arg), UNUSED_PAR(void* (*f)(void*) noexcept)) noexcept
406  {
407  // clang-format off
408  asm volatile(
409  "push {r4, " REG_FP ", lr}\n" // Save pc and variables
410  ".save {r4, " REG_FP ", lr}\n"
411  "add " REG_FP ", sp, #0\n"
412 
413  "mrs r4, control\n" // Save current control register
414  "bic r3, r4, #2\n" // Set to use MSP
415  "msr control, r3\n" // Enable MSP
416  "isb\n"
417  "and r4, r4, #2\n" // r4 is set to only have the PSP bit
418 
419  // We are on MSP now.
420 
421  "push {" REG_FP ", lr}\n" // Save frame pointer on MSP
422  ".setfp " REG_FP ", sp\n"
423  "add " REG_FP ", sp, #0\n"
424 
425  "blx r1\n" // Call f(arg)
426 
427  "mrs r3, control\n" // Save current control register
428  "orr r3, r3, r4\n" // Set to use previous PSP setting
429  "msr control, r3\n" // Enable PSP setting
430  "isb\n"
431 
432  // We are back on the previous stack.
433 
434  "pop {r4, " REG_FP ", pc}\n" // Return to caller
435  );
436  // clang-format on
437  }

465  {
466  if(!stack) {

468 
469  if(worker)

472  void* res = stack_switch_msp(arg, f);
473 
474  if(worker)

477  return res;
