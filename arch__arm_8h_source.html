151 static int stackGuardRegion() noexcept
152 {
153 if(!Config::EnableStackGuard || Config::EnableThreads)
154 return -1;
155
156 static int region = 0;
157
158 if(likely(region))
159 return region;
160
161 region = reg_mpu_type().field.dregion - 1;
162 reg_mpu_rnr rnr;
163 reg_mpu_rasr rasr;
164
165 // Do not try to use region 0, leave at least one unused for other application
166 // purposes.
167 for(; region > 0; --region) {
168 rnr.field.region = region;
169 rnr.write();
170
171 rasr.read();
172 if(!rasr.field.enable)
173 // This one is free.
174 break;
175 }
176
177 if(region > 0) {
178 reg_mpu_rbar rbar;
179 // Initialize at the top of the address space, so it is effectively unused.
180 rbar.field.addr = REG_MPU_RBAR_ADDR_UNUSED;
181 rbar.write();
182
183 rasr.field.size = ZTH_ARM_STACK_GUARD_BITS - 1;
184 rasr.field.xn = 0;
185 rasr.field.ap = 0; // no access
186
187 // Internal RAM should be normal, shareable, write-through.
188 rasr.field.tex = 0;
189 rasr.field.c = 1;
190 rasr.field.b = 0;
191 rasr.field.s = 1;
192
193 rasr.field.enable = 1;
194 rasr.write();
195
196 reg_mpu_ctrl ctrl;
197 if(!ctrl.field.enable) {
198 // MPU was not enabled at all.
199 ctrl.field.privdefena = 1; // Assume we are currently privileged.
200 ctrl.field.enable = 1;
201 ctrl.write();
202 __isb();
203 }
204 zth_dbg(context, "Using MPU region %d as stack guard", region);
205 } else {
206 region = -1;
207 zth_dbg(context, "Cannot find free MPU region for stack guard");
208 }
209
210 return region;
211 }
212
213 static void stackGuardImpl(void* guard) noexcept
214 {
215 if(!Config::EnableStackGuard || Config::EnableThreads)
216 return;
217
218 int const region = stackGuardRegion();
219 if(unlikely(region < 0))
220 return;
221
222 // Must be aligned to guard size
223 zth_assert(((uintptr_t)guard & (ZTH_ARM_STACK_GUARD_SIZE - 1)) == 0);
224
225 reg_mpu_rbar rbar(0);
226
227 if(likely(guard)) {
228 rbar.field.addr = (uintptr_t)guard >> 5;
229 zth_dbg(context, "Set MPU stack guard to %p (sp=%p)", guard, sp());
230 } else {
231 // Initialize at the top of the address space, so it is effectively unused.
232 rbar.field.addr = REG_MPU_RBAR_ADDR_UNUSED;
233 zth_dbg(context, "Disabled MPU stack guard");
234 }
235
236 rbar.field.valid = 1;
237 rbar.field.region = region;
238 rbar.write();
239
240 if(Config::Debug) {
241 // Only required if the settings should be in effect immediately.
242 // However, it takes some time, which may not be worth it in a release
243 // build. If skipped, it will take affect, but only after a few
244 // instructions, which are already in the pipeline.
245 __dsb();
246 __isb();
247 }
248 }

290private:
291 static size_t get_lr_offset() noexcept
292 {
293 static size_t lr_offset = 0;
294
295 if(unlikely(lr_offset == 0)) {
296 // Not initialized yet.
297
298 // env is filled with several registers. The number of registers depends on
299 // the ARM architecture and Thumb mode, and compile settings for newlib. We
300 // have to change the sp and lr registers in this env. These follow other
301 // caller-saved registers, but may be placed before VPU registers.
302 //
303 // Instead of hardcoding the offset, we try to find it dynamically. The sp
304 // and lr registers are always next to each other. We assume here that the
305 // sp is not part of the VPU registers, so this is our marker.
306
307 static_assert(
308 sizeof(jmp_buf) >= sizeof(void*) * 4U,
309 "jmp_buf size too small to find lr offset");
310
311 jmp_buf test_env = {};
312 ::setjmp(test_env);
313
314 uintptr_t sp_ = 0;
315 asm volatile("mov %0, sp;" : "=r"(sp_));
316 zth_assert(sp_);
317
318 // Note that jmp_buf may be an array type with other elements than void*.
319 // NOLINTNEXTLINE
320 uintptr_t const* test_env_ = reinterpret_cast<uintptr_t const*>(test_env);
321# pragma GCC diagnostic push
322# pragma GCC diagnostic ignored "-Wsizeof-array-div"
323 size_t sp_offset = sizeof(jmp_buf) / sizeof(void*) - 2U;
324# pragma GCC diagnostic pop
325 for(; sp_offset > 0 && test_env_[sp_offset] != sp_; sp_offset--)
326 ;
327
328 if(sp_offset == 0) {
329 // Should not be possible.
330 zth_abort("Invalid lr offset");
331 }
332
333 lr_offset = sp_offset + 1U;
334 }
335
336 return lr_offset;
337 }

408# ifdef ZTH_ARM_USE_PSP
409 inline __attribute__((always_inline)) void
410 context_prepare_jmp(Impl& to, jmp_buf& env) noexcept
411 {
412 // Use the PSP for fibers and the MSP for the worker.
413 // So, if there is no stack in the Context, assume it is a worker, running
414 // from MSP. Otherwise it is a fiber using PSP. Both are executed privileged.
415 if(unlikely(!impl().stackUsable() || !to.stackUsable())) {
416 unsigned int control;
417 asm("mrs %0, control\n" : "=r"(control));
418
419 if(to.stackUsable())
420 control |= 2U; // set SPSEL to PSP
421 else
422 control &= ~2U; // set SPSEL to MSR
423
424 // As the SP changes, do not return from this function,
425 // but actually do the longjmp.
426 asm volatile(
427 "msr control, %1\n"
428 "isb\n"
429 "mov r0, %0\n"
430 "movs r1, #1\n"
431 "b longjmp\n"
432 :
433 : "r"(env), "r"(control)
434 : "r0", "r1", "memory");
435 }
436 }
437# endif
438# endif // ZTH_CONTEXT_SJLJ
439# endif // ZTH_OS_BAREMETAL

442private:
443 __attribute__((naked, noinline)) static void*
444 stack_switch_msp(void* UNUSED_PAR(arg), UNUSED_PAR(void* (*f)(void*) noexcept)) noexcept
445 {
446 // clang-format off
447 asm volatile(
448 "push {r4, " REG_FP ", lr}\n" // Save pc and variables
449#ifdef __cpp_exceptions
450 ".save {r4, " REG_FP ", lr}\n"
451#endif
452 "add " REG_FP ", sp, #0\n"
453
454 "mrs r4, control\n" // Save current control register
455 "bic r3, r4, #2\n" // Set to use MSP
456 "msr control, r3\n" // Enable MSP
457 "isb\n"
458 "and r4, r4, #2\n" // r4 is set to only have the PSP bit
459
460 // We are on MSP now.
461
462 "push {" REG_FP ", lr}\n" // Save frame pointer on MSP
463#ifdef __cpp_exceptions
464 ".setfp " REG_FP ", sp\n"
465#endif
466 "add " REG_FP ", sp, #0\n"
467
468 "blx r1\n" // Call f(arg)
469
470 "mrs r3, control\n" // Save current control register
471 "orr r3, r3, r4\n" // Set to use previous PSP setting
472 "msr control, r3\n" // Enable PSP setting
473 "isb\n"
474
475 // We are back on the previous stack.
476
477 "pop {r4, " REG_FP ", pc}\n" // Return to caller
478 );
479 // clang-format on
480 }

482 __attribute__((naked, noinline)) static void* stack_switch_psp(
483 void* UNUSED_PAR(arg), UNUSED_PAR(void* (*f)(void*) noexcept),
484 void* UNUSED_PAR(sp)) noexcept
485 {
486 // clang-format off
487 asm volatile(
488 "push {r4, " REG_FP ", lr}\n" // Save pc and variables
489#ifdef __cpp_exceptions
490 ".save {r4, " REG_FP ", lr}\n"
491#endif
492 "add " REG_FP ", sp, #0\n"
493
494 "mov r4, sp\n" // Save previous stack pointer
495 "mov sp, r2\n" // Set new stack pointer
496 "push {" REG_FP ", lr}\n" // Save previous frame pointer on new stack
497#ifdef __cpp_exceptions
498 ".setfp " REG_FP ", sp\n"
499#endif
500 "add " REG_FP ", sp, #0\n"
501
502 "blx r1\n" // Call f(arg)
503
504 "mov sp, r4\n" // Restore previous stack
505 "pop {r4, " REG_FP ", pc}\n" // Return to caller
506 );
507 // clang-format on
508 }

510public:
511 void* stack_switch(void* stack, size_t size, void* (*f)(void*) noexcept, void* arg) noexcept
512 {
513 if(!stack) {
514 zth::Worker* worker = zth::Worker::instance();
515
516 if(worker)
517 worker->contextSwitchDisable();
518
519 void* res = stack_switch_msp(arg, f);
520
521 if(worker)
522 worker->contextSwitchEnable();
523
524 return res;
525 } else {
526 void* sp = (void*)(((uintptr_t)stack + size) & ~(uintptr_t)7);
527# ifdef ZTH_ARM_HAVE_MPU
528 void* prevGuard = nullptr;
529
530 if(zth::Config::EnableStackGuard && size >= ZTH_ARM_STACK_GUARD_SIZE * 2) {
531 void* guard = (void*)(((uintptr_t)stack
532 + 2 * ZTH_ARM_STACK_GUARD_SIZE - 1)
533 & ~(ZTH_ARM_STACK_GUARD_SIZE - 1));
534 prevGuard = setGuard(guard);
535 stackGuard();
536 }
537# endif
538
539 void* res = stack_switch_psp(arg, f, sp);
540
541# ifdef ZTH_ARM_HAVE_MPU
542 if(prevGuard) {
543 setGuard(prevGuard);
544 stackGuard();
545 }
546# endif
547 return res;
548 }
549 }
550# endif // ZTH_STACK_SWITCH
