441  int remove(Pollable& p, Client* client) noexcept final
442  {
443  size_t count = m_metaItems.size();
444 
445  size_t i;
446  for(i = count; i > 0; i--) {
447  MetaItem& m = m_metaItems[i - 1u];
448  if(m.pollable == &p && m.client == client)
449  break;
450  }
451 
452  if(i == 0u)
453  return ESRCH;
454 
455  i--;
456 
457  if(i < count - 1u) {
458  // Not removing the last element, fill the gap.
459  if(i < m_pollItems.size()) {
460  deinit(*m_metaItems[i].pollable, m_pollItems[i]);
461  if(i < m_pollItems.size() - 1u) {
462 #if __cplusplus >= 201103L
463  m_pollItems[i] = std::move(m_pollItems.back());
464 #else
465  m_pollItems[i] = m_pollItems.back();
466 #endif
467  }
468  m_pollItems.pop_back();
469  }
470 
471 #if __cplusplus >= 201103L

473 #else

477  } else {

483  zth_assert(m_metaItems.size() >= m_pollItems.size());

764  int poll(P pollable, int timeout_ms = -1)
765  {
766  __try {
767  Poller poller;
768  poller.add(pollable);
769 
770  while(true) {
771  Poller::Result const& result = poller.poll(timeout_ms);
772 
773  if(result.empty()) {
774  switch(errno) {
775  case EINTR:
776  case EAGAIN:
777  if(timeout_ms == -1)
778  // Retry.
779  continue;
780  ZTH_FALLTHROUGH
781  default:
782  // Error.
783  return errno;
784  }
785  }
786 
787  if((result[0]->revents & result[0]->events).any()) {
788  // Got it.
789  return 0;
790  }
791 
792  // Hmm, there is another reason we returned...
793  return EIO;
794  }
795  } __catch(...) {
796  return ENOMEM;
797  }
