447  int remove(Pollable& p, Client* client) noexcept final
448  {
449  size_t count = m_metaItems.size();
450 
451  size_t i;
452  for(i = count; i > 0; i--) {
453  MetaItem& m = m_metaItems[i - 1u];
454  if(m.pollable == &p && m.client == client)
455  break;
456  }
457 
458  if(i == 0u)
459  return ESRCH;
460 
461  i--;
462 
463  if(i < count - 1u) {
464  // Not removing the last element, fill the gap.
465  if(i < m_pollItems.size()) {
466  deinit(*m_metaItems[i].pollable, m_pollItems[i]);
467  if(i < m_pollItems.size() - 1u) {
468 # if __cplusplus >= 201103L
469  m_pollItems[i] = std::move(m_pollItems.back());
470 # else
471  m_pollItems[i] = m_pollItems.back();
472 # endif
473  }
474  m_pollItems.pop_back();
475  }
476 
477 # if __cplusplus >= 201103L
478  m_metaItems[i] = std::move(m_metaItems.back());
479 # else
480  m_metaItems[i] = m_metaItems.back();
481 # endif
482  m_metaItems.pop_back();
483  } else {
484  // Drop the last element.
485  remove(1u);
486  }
487 
488  zth_dbg(io, "[%s] removed pollable %p", this->id_str(), &p);
489  zth_assert(m_metaItems.size() >= m_pollItems.size());
490  return 0;
491  }

497  {

788 template <typename P>
789 int poll(P pollable, int timeout_ms = -1)
790 {
791  try {
792  Poller poller;
793  poller.add(pollable);
794 
795  while(true) {
796  Poller::Result const& result = poller.poll(timeout_ms);
797 
798  if(result.empty()) {
799  switch(errno) {
800  case EINTR:
801  case EAGAIN:
802  if(timeout_ms == -1)
803  // Retry.
804  continue;
805  ZTH_FALLTHROUGH
806  default:
807  // Error.
808  return errno;
809  }
810  }
811 
812  if((result[0]->revents & result[0]->events).any()) {
813  // Got it.
814  return 0;
815  }
816 
817  // Hmm, there is another reason we returned...
818  return EIO;
819  }
820  } catch(...) {
821  return ENOMEM;
822  }
823 }
