39 static void* zmq_global_init()
40 {
41  int major = 0;
42  int minor = 0;
43  int patch = 0;
44  zmq_version(&major, &minor, &patch);
45  zth_dbg(banner, "0MQ version is %d.%d.%d", major, minor, patch);
46 
47  zth_dbg(zmq, "new context");
48  void* zmq_ctx = zmq_ctx_new();
49 
50  if(!zmq_ctx)
51  zth_abort("0MQ context creation failed; %s", err(errno).c_str());
52 
53  // Only do the deinit this when 0MQ was actually used.
54  atexit(zmq_global_deinit);
55  return zmq_ctx;
56 }

82 int zmq_msg_send(zmq_msg_t* msg, void* socket, int flags)
83 {
84  perf_syscall("zmq_msg_send()");
85 
86  int res = ::zmq_msg_send(msg, socket, flags | ZMQ_DONTWAIT);
87  int err = res == -1 ? zmq_errno() : 0;
88  if(err != EAGAIN || (flags & ZMQ_DONTWAIT))
89  return res;
90 
91  zth_dbg(zmq, "[%s] zmq_msg_send(%p) hand-off", zth::currentFiber().str().c_str(), socket);
92 
93  if((errno = poll(PollableFd(socket, Pollable::PollOut))))
94  return -1;
95 
96  return ::zmq_msg_send(msg, socket, flags | ZMQ_DONTWAIT);
