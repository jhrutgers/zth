37template <typename R, typename F>
38class TypedFiber : public Fiber {
39 ZTH_CLASS_NEW_DELETE(TypedFiber)
40public:
41 typedef R Return;
42 typedef F Function;
43 typedef Future<Return> Future_type;
44
45 explicit TypedFiber(Function func)
46 : Fiber(&entry, this)
47 , m_function(func)
48 {}
49
50 virtual ~TypedFiber() override is_default
51
52 void registerFuture(Future_type* future)
53 {
54 m_future.reset(future);
55 zth_dbg(sync, "[%s] Registered to %s", future->id_str(), id_str());
56 }
57
58 Future_type* future() const
59 {
60 return m_future;
61 }
62
63 SharedPointer<Future_type> withFuture()
64 {
65 if(!m_future.get()) {
66 registerFuture(
67 !Config::NamedSynchronizer
68 ? new Future_type()
69 : new Future_type(
70 format("Future of %s", this->name().c_str())
71 .c_str()));
72 }
73
74 return m_future;
75 }
76
77protected:
78 static void entry(void* that)
79 {
80 if(likely(that))
81 static_cast<TypedFiber*>(that)->entry_();
82 }
83
84 virtual void entry_() = 0;
85
86 template <typename T>
87 void setFuture(T const& r)
88 {
89 if(m_future.get())
90 m_future->set(r);
91 }
92
93 void setFuture()
94 {
95 if(m_future.get())
96 m_future->set();
97 }
98
99# ifdef ZTH_FUTURE_EXCEPTION
100 void setFuture(std::exception_ptr exception)
101 {
102 if(m_future.get())
103 m_future->set(std::move(exception));
104 }
105# endif // ZTH_FUTURE_EXCEPTION
106
107 Function function() const noexcept
108 {
109 return m_function;
110 }
111
112private:
113 Function m_function;
114 SharedPointer<Future_type> m_future;
115};
