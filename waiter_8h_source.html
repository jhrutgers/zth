123 class PolledWaiting : public TimedWaitable {
124  ZTH_CLASS_NEW_DELETE(PolledWaiting)
125 public:
126  explicit PolledWaiting(F const& f, TimeInterval const& interval = TimeInterval())
127  : TimedWaitable(Timestamp())
128  , m_f(f)
129  {
130  setInterval(interval);
131  }
132 
133  virtual ~PolledWaiting() override is_default
134 
135  virtual bool poll(Timestamp const& now = Timestamp::now()) noexcept override
136  {
137  if(m_f())
138  return true;
139 
140  Timestamp next = timeout() + interval();
141  if(unlikely(next < now))
142  next = now + interval();
143 
144  setTimeout(next);
145  return false;
146  }
147 
148  TimeInterval const& interval() const noexcept
