70  virtual bool poll(Timestamp const& now = Timestamp::now()) override {
71  if(m_f())
72  return true;
73 
74  Timestamp next = timeout() + interval();
75  if(unlikely(next < now))
76  next = now + interval();
77 
78  setTimeout(next);
79  return false;
80  }
