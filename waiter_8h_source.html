87  virtual bool poll(Timestamp const& now = Timestamp::now()) {
88  if(m_f())
89  return true;
90 
91  Timestamp next = timeout() + interval();
92  if(unlikely(next < now))
93  next = now + interval();
94 
95  setTimeout(next);
96  return false;
97  }
