67 int context_create(Context*& context, ContextAttr const& attr) noexcept
68 {
69  try {
70  context = new Context(attr);
71  } catch(std::bad_alloc const&) {
72  zth_dbg(context, "[%s] Cannot create context; %s", currentWorker().id_str(),
73  err(ENOMEM).c_str());

75  } catch(...) {
76  // Should not be thrown by new...

81

83  delete context;
84  context = nullptr;
85  zth_dbg(context, "[%s] Cannot create context; %s", currentWorker().id_str(),
86  err(res).c_str());
87  return res;
88  }
89 
90  if(likely(attr.stackSize > 0)) {
91 #ifdef ZTH_CONTEXT_WINFIBER
92  zth_dbg(context, "[%s] New context %p", currentWorker().id_str(), context);
93 #else
94  Context::Stack const& stack = context->stackUsable();
95  zth_dbg(context, "[%s] New context %p with stack: %p-%p", currentWorker().id_str(),
96  context, stack.p, stack.p + stack.size - 1u);
97 #endif
98  }
99 
100  return 0;
101 }

126 {

134  from->alive() ? &fake_stack : nullptr, to->stackUsable().p, to->stackUsable().size);
135 #endif
136 
137  from->context_switch(*to);
138 
139 #ifdef ZTH_ENABLE_ASAN
140  __sanitizer_finish_switch_fiber(fake_stack, nullptr, nullptr);
141 #endif
142 
143  // Got back from somewhere else.
144  from->stackGuard();
145 }

152 {

179 {

182 #endif
183 
184  zth_assert(context);
185 
186  // Go execute the fiber.
187  context->stackGuard();
188 
189  try {
190  context->attr().entry(context->attr().arg);
191  } catch(...) {
192  }
193 
194  // Fiber has quit. Switch to another one.
195  context->die();

199  zth_abort("Returned to finished context");

218 static void stack_watermark_align(void*& stack, size_t*& sizeptr, size_t* size = nullptr) noexcept
219 {
220  if(!stack)
221  return;
222 
223  uintptr_t stack_ = reinterpret_cast<uintptr_t>(stack);
224 
225  // Align to size_t
226  stack_ = ((uintptr_t)stack_ + sizeof(size_t) - 1) & ~(uintptr_t)(sizeof(size_t) - 1);
227  // The stack size is stored here.
228  sizeptr = reinterpret_cast<size_t*>(stack_);
229 
230  // Reduce stack to store the size.
231  stack_ += sizeof(size_t);
232  if(size)
233  // Reduce remaining stack size.
234  *size -= stack_ - reinterpret_cast<uintptr_t>(stack);
235 
236  stack = reinterpret_cast<void*>(stack_);
237 }

243 void stack_watermark_init(void* stack, size_t size) noexcept
244 {
245  if(!Config::EnableStackWaterMark || !stack || !size)
246  return;
247 
248  // Most likely, but growing up stack is not implemented here (yet?)
249  zth_assert(Context::stackGrowsDown(&size));
250 
251  size_t* sizeptr = nullptr;
252  stack_watermark_align(stack, sizeptr, &size);
253  *sizeptr = size;
254 
255  for(size_t i = 0; i * sizeof(uint64_t) <= size - sizeof(uint64_t); i++)
256  static_cast<uint64_t*>(stack)[i] = ZTH_STACK_WATERMARKER;
257 }

281 size_t stack_watermark_maxused(void* stack) noexcept
282 {
283  if(!Config::EnableStackWaterMark || !stack)
284  return 0;
285 
286  size_t* sizeptr = nullptr;
287  stack_watermark_align(stack, sizeptr);
288  size_t size = *sizeptr;
289 
290  size_t unused = 0;
291 
292 #ifdef ZTH_USE_VALGRIND
293  VALGRIND_DISABLE_ADDR_ERROR_REPORTING_IN_RANGE(stack, size); // NOLINT
294 #endif
295 
296  for(uint64_t* s = static_cast<uint64_t*>(stack);
297  unused < size / sizeof(uint64_t) && *s == ZTH_STACK_WATERMARKER; unused++, s++)
298  ;
299 
300 #ifdef ZTH_USE_VALGRIND
301  VALGRIND_ENABLE_ADDR_ERROR_REPORTING_IN_RANGE(stack, size); // NOLINT
302 #endif
303 
304  return size - unused * sizeof(uint64_t);
305 }
