154 #pragma GCC diagnostic ignored "-Wconversion"
155 static int stack_guard_region()
156 {
157  if(!Config::EnableStackGuard || Config::EnableThreads)
158  return -1;
159 
160  static int region = 0;
161 
162  if(likely(region))
163  return region;
164 
165  region = reg_mpu_type().field.dregion - 1;
166  reg_mpu_rnr rnr;
167  reg_mpu_rasr rasr;
168 
169  // Do not try to use region 0, leave at least one unused for other application purposes.
170  for(; region > 0; --region) {
171  rnr.field.region = region;
172  rnr.write();
173 
174  rasr.read();
175  if(!rasr.field.enable)
176  // This one is free.
177  break;
178  }
179 
180  if(region > 0) {
181  reg_mpu_rbar rbar;
182  // Initialize at the top of the address space, so it is effectively unused.
183  rbar.field.addr = REG_MPU_RBAR_ADDR_UNUSED;
184  rbar.write();
185 
186  rasr.field.size = ZTH_ARM_STACK_GUARD_BITS - 1;
187  rasr.field.xn = 0;
188  rasr.field.ap = 0; // no access
189 
190  // Internal RAM should be normal, shareable, write-through.
191  rasr.field.tex = 0;
192  rasr.field.c = 1;
193  rasr.field.b = 0;
194  rasr.field.s = 1;
195 
196  rasr.field.enable = 1;
197  rasr.write();
198 
199  reg_mpu_ctrl ctrl;
200  if(!ctrl.field.enable) {
201  // MPU was not enabled at all.
202  ctrl.field.privdefena = 1; // Assume we are currently privileged.
203  ctrl.field.enable = 1;
204  ctrl.write();
205  __isb();
206  }
207  zth_dbg(context, "Using MPU region %d as stack guard", region);
208  } else {
209  region = -1;
210  zth_dbg(context, "Cannot find free MPU region for stack guard");
211  }
212 
213  return region;
214 }

216 static void stack_guard(void* guard)
217 {
218  if(!Config::EnableStackGuard || Config::EnableThreads)
219  return;
220 
221  int const region = stack_guard_region();
222  if(unlikely(region < 0))
223  return;
224 
225  // Must be aligned to guard size
226  zth_assert(((uintptr_t)guard & (ZTH_ARM_STACK_GUARD_SIZE - 1)) == 0);
227 
228  reg_mpu_rbar rbar(0);
229 
230  if(likely(guard)) {
231  rbar.field.addr = (uintptr_t)guard >> 5;
232  register void* sp asm ("sp");
233  zth_dbg(context, "Set MPU stack guard to %p (sp=%p)", guard, sp);
234  } else {
235  // Initialize at the top of the address space, so it is effectively unused.
236  rbar.field.addr = REG_MPU_RBAR_ADDR_UNUSED;
237  zth_dbg(context, "Disabled MPU stack guard");
238  }
239 
240  rbar.field.valid = 1;
241  rbar.field.region = region;
242  rbar.write();
243 
244  if(Config::Debug) {
245  // Only required if the settings should be in effect immediately.
246  // However, it takes some time, which may not be worth it in a release build.
247  // If skipped, it will take affect, but only after a few instructions, which
248  // are already in the pipeline.
249  __dsb();
250  __isb();
251  }
252 }

256 # define stack_guard(...)

357 static void bootstrap(Context* context) __attribute__((noreturn
358 #ifdef ZTH_USE_VALGRIND
359  ,noinline
360 #endif
361  ));
362 static void bootstrap(Context* context)
363 {
364  zth_dbg(context, "Bootstrapping %p", context);
365 
366  if(Config::ContextSignals) {
367  int res = pthread_sigmask(SIG_SETMASK, &context->mask, nullptr);
368  if(unlikely(res))
369  zth_abort("Cannot set signal mask");
370  }
371 
372  // However, we still need to setup the context for the first time.
373  if(sigsetjmp(context->env, Config::ContextSignals) == 0) {
374  // Now, we are ready to be context switched to this fiber.
375  // Go back to our parent, as it was not our schedule turn to continue with actual execution.
376  siglongjmp(*context->parent, 1);
377  }
378 
379  // This is actually the first time we enter the fiber by the normal scheduler.
380  context_entry(context);
381 }

385 static void context_trampoline(int sig)
386 {
387  // At this point, we are in the signal handler.
388 
389  if(unlikely(sig != SIGUSR1))
390  // Huh?
391  return;
392 
393  // Put the context on the stack, as trampoline_context is about to change.
394  Context* context = ZTH_TLS_GET(trampoline_context);
395 
396  if(unlikely(!context || context->did_trampoline))
397  // Huh?
398  return;
399 
400  if(setjmp(context->trampoline_env) == 0)
401  {
402  // Return immediately to complete the signal handler.
403  context->did_trampoline = 1;
404  return;
405  }
406 
407  // If we get here, we are not in the signal handler anymore, but in the actual fiber context.
408  bootstrap(context);
409 }

441 static int context_create_impl(Context* context, stack_t* stack)
442 {
443  if(unlikely(!stack->ss_sp))
444  // Stackless fiber only saves current context; nothing to do.
445  return 0;
446 
447  int res = 0;
448  context->did_trampoline = 0;
449 
450  // Let the new context inherit our signal mask.
451  if(Config::ContextSignals)
452  if(unlikely((res = pthread_sigmask(0, nullptr, &context->mask))))
453  return res;
454 
455  stack_t& ss = *stack;
456  stack_t oss;
457 
458 #ifdef ZTH_USE_VALGRIND
459  // Disable checking during bootstrap.
460  VALGRIND_DISABLE_ADDR_ERROR_REPORTING_IN_RANGE(ss.ss_sp, ss.ss_size);
461 #endif
462 
463 #ifdef SS_AUTODISARM
464  ss.ss_flags |= SS_AUTODISARM;
465 #endif
466  if(unlikely(sigaltstack(&ss, &oss)))
467  return errno;
468 
469  ZTH_TLS_SET(trampoline_context, context);
470 
471  // Ready to do the signal.
472  sigset_t sigs;
473  sigemptyset(&sigs);
474  sigaddset(&sigs, SIGUSR1);
475  if(unlikely((res = pthread_sigmask(SIG_UNBLOCK, &sigs, nullptr))))
476  goto rollback_altstack;
477 
478  // As we are not blocking SIGUSR1, we will get the signal.
479  if(unlikely((res = pthread_kill(pthread_self(), SIGUSR1))))
480  goto rollback_altstack;
481 
482  // Shouldn't take long...
483  while(!context->did_trampoline)
484  pthread_yield_np();
485 
486  // Block again.
487  if(unlikely((res = pthread_sigmask(SIG_BLOCK, &sigs, nullptr))))
488  goto rollback_altstack;
489 
490  if(Config::Debug)
491  ZTH_TLS_SET(trampoline_context, nullptr);
492 
493  if(Config::EnableAssert) {
494  if(unlikely(sigaltstack(nullptr, &ss))) {
495  res = errno;
496  goto rollback_altstack;
497  }
498  zth_assert(!(ss.ss_flags & SS_ONSTACK));
499  }
500 
501 #ifndef SS_AUTODISARM
502  // Reset sigaltstack.
503  ss.ss_flags = SS_DISABLE;
504  if(unlikely(sigaltstack(&ss, nullptr))) {
505  res = errno;
506  goto rollback_altstack;
507  }
508 #endif
509 
510  if(unlikely(!(oss.ss_flags & SS_DISABLE))) {
511  if(sigaltstack(&oss, nullptr)) {
512  res = errno;
513  goto rollback;
514  }
515  }
516 #ifdef ZTH_HAVE_VALGRIND
517  else if(RUNNING_ON_VALGRIND) {
518  // Valgrind has a hard time tracking when we left the signal handler
519  // and are not using the altstack anymore. Reset the altstack, such
520  // that successive sigaltstack() do not fail with EPERM.
521  stack_t dss;
522  dss.ss_sp = dummyAltStack;
523  dss.ss_size = sizeof(dummyAltStack);
524  dss.ss_flags = 0;
525  if(unlikely(sigaltstack(&dss, nullptr))) {
526  res = errno;
527  goto rollback;
528  }
529 
530  // Although the previous stack was disabled, it is still valid.
531  VALGRIND_MAKE_MEM_DEFINED(ss.ss_sp, ss.ss_size);
532  }
533 #endif
534 
535  // Ok, we have returned from the signal handler.
536  // Now go back again and save a proper env.
537 
538  sigjmp_buf me;
539  context->parent = &me;
540  if(sigsetjmp(me, Config::ContextSignals) == 0)
541  longjmp(context->trampoline_env, 1);
542 
543 #ifdef ZTH_USE_VALGRIND
544  // Disabled checking during bootstrap.
545  VALGRIND_ENABLE_ADDR_ERROR_REPORTING_IN_RANGE(ss.ss_sp, ss.ss_size);
546 #endif
547 
548  if(Config::EnableAssert) {
549  if(unlikely(sigaltstack(nullptr, &oss))) {
550  res = errno;
551  goto rollback;
552  }
553  zth_assert(!(oss.ss_flags & SS_ONSTACK));
554  }
555 
556  // context is setup properly and is ready for normal scheduling.
557  if(Config::Debug)
558  context->parent = nullptr;
559 
560  return 0;
561 
562 rollback_altstack:
563 #ifdef ZTH_USE_VALGRIND
564  VALGRIND_ENABLE_ADDR_ERROR_REPORTING_IN_RANGE(ss.ss_sp, ss.ss_size);
565 #endif
566  if(!(oss.ss_flags & SS_DISABLE))
567  sigaltstack(&oss, nullptr);
568 rollback:
569  return res ? res : EINVAL;
570 }

705 static void context_switch_impl(Context* from, Context* to)
706 {
707 #if defined(__ARM_PCS_VFP) && !defined(__SOFTFP__)
708  // newlib does not do saving/restoring of FPU registers.
709  // We have to do it here ourselves.
710  __asm__ volatile ("vpush {d8-d15}");
711  #define arm_fpu_pop() __asm__ volatile ("vpop {d8-d15}")
712 #else
713  #define arm_fpu_pop() do{}while(0)
714 #endif
715 
716  // As bare metal does not have signals, sigsetjmp and siglongjmp is not necessary.
717  if(setjmp(from->env) == 0)
718  {
719 #ifdef ZTH_ARM_USE_PSP
720  // Use the PSP for fibers and the MSP for the worker.
721  // So, if there is no stack in the Context, assume it is a worker, running
722  // from MSP. Otherwise it is a fiber using PSP. Both are executed privileged.
723  if(unlikely(!from->stack || !to->stack)) {
724  int control;
725  __asm__ ("mrs %0, control\n" : "=r"(control));
726 
727  if(to->stack)
728  control |= 2; // set SPSEL to PSP
729  else
730  control &= ~2; // set SPSEL to MSR
731

780 static int context_newstack(Context* context, stack_t* stack)
781 {

783  int res = 0;
784  size_t const pagesize =
785 #ifdef ZTH_HAVE_MMAN
786  (size_t)getpagesize();
787 #elif defined(ZTH_ARM_HAVE_MPU)
788  Config::EnableStackGuard ? (size_t)ZTH_ARM_STACK_GUARD_SIZE : sizeof(void*);
789 #else
790  sizeof(void*);
791 #endif
792  zth_assert(__builtin_popcount((unsigned int)pagesize) == 1);
793  context->stackSize = (context->stackSize
794 #ifndef ZTH_STACK_SWITCH
795  // If using stack switch, signals and interrupts are not executed on the fiber stack.
796  + MINSIGSTKSZ
797 #endif
798  + 3 * pagesize - 1) & ~(pagesize - 1);
799 
800 
801  // Get a new stack.
802  size_t stack_watermarked_size = 0;
803 
804 #ifdef ZTH_HAVE_MMAN
805  // NOLINTNEXTLINE(hicpp-signed-bitwise,cppcoreguidelines-pro-type-cstyle-cast)
806  if(unlikely((context->stack = mmap(nullptr, context->stackSize, PROT_READ | PROT_WRITE, MAP_PRIVATE | MAP_ANONYMOUS | MAP_STACK, -1, 0)) == MAP_FAILED))
807 #else
808  if(unlikely((context->stack = malloc(context->stackSize)) == nullptr))
809 #endif
810  {
811  res = errno;
812  context->stack = nullptr;
813  goto rollback;
814  }
815 
816 #ifdef ZTH_HAVE_MMAN
817  if(Config::EnableStackGuard) {
818  // Exclude the pages at both sides of the stack.
819  context->stack_watermarked = stack->ss_sp = static_cast<char*>(context->stack) + pagesize;
820  stack_watermarked_size = stack->ss_size = context->stackSize - 2 * pagesize;
821  } else
822 #elif defined(ZTH_ARM_HAVE_MPU)
823  if(Config::EnableStackGuard) {
824  // Stack grows down, exclude the page at the end (lowest address).
825  // The page must be aligned to its size.
826  uintptr_t stackEnd = ((uintptr_t)context->stack + 2 * pagesize - 1) & ~(pagesize - 1);
827  stack->ss_sp = (char*)stackEnd;
828  stack->ss_size = context->stackSize - (stackEnd - (uintptr_t)context->stack);
829  /* context->stack_watermarked = */ context->guard = (void*)(stackEnd - ZTH_ARM_STACK_GUARD_SIZE);
830  stack_watermarked_size = stack->ss_size + ZTH_ARM_STACK_GUARD_SIZE;
831  } else
832 #endif
833  {
834  context->stack_watermarked = stack->ss_sp = context->stack;
835  stack_watermarked_size = stack->ss_size = context->stackSize;
836  }
837  stack->ss_flags = 0;
838 
839  if(Config::EnableStackWaterMark)
840  stack_watermark_init(context->stack_watermarked, stack_watermarked_size);
841  else
842  context->stack_watermarked = nullptr;
843 
844 #ifdef ZTH_USE_VALGRIND
845  context->valgrind_stack_id = VALGRIND_STACK_REGISTER(stack->ss_sp, (char*)stack->ss_sp + stack->ss_size - 1);
846 
847  if(RUNNING_ON_VALGRIND)
848  zth_dbg(context, "[%s] Stack of context %p has Valgrind id %u",
849  currentWorker().id_str(), context, context->valgrind_stack_id);
850 #endif
851 
852 #ifdef ZTH_HAVE_MMAN
853  if(Config::EnableStackGuard) {
854  // Guard both ends of the stack.
855  if(unlikely(
856  mprotect(context->stack, pagesize, PROT_NONE) ||
857  mprotect(static_cast<char*>(context->stack) + context->stackSize - pagesize, pagesize, PROT_NONE)))
858  {
859  res = errno;
860  goto rollback_mmap;
861  }
862  }
863 #endif
864

866

868 rollback_mmap:
