179  virtual ~Synchronizer() override

185 protected:
186  void block()
187  {
188  Worker* w = nullptr;
189  Fiber* f = nullptr;
190  getContext(&w, &f);
191 
192  zth_dbg(sync, "[%s] Block %s", id_str(), f->id_str());
193  w->release(*f);
194  m_queue.push_back(*f);
195  f->nap(Timestamp::null());
196  w->schedule();
197  }
198 
203  bool unblock(Fiber& f) noexcept
204  {
205  if(!m_queue.contains(f))
206  return false;
207 
208  Worker* w = nullptr;
209  getContext(&w, nullptr);
210 
211  zth_dbg(sync, "[%s] Unblock %s", id_str(), f.id_str());
212  m_queue.erase(f);
213  f.wakeup();
214  w->add(&f);
215  return true;
216  }
217 
218  bool unblockFirst() noexcept
219  {
220  if(m_queue.empty())
221  return false;
222 
223  Worker* w = nullptr;
224  getContext(&w, nullptr);
225 
226  Fiber& f = m_queue.front();
227  zth_dbg(sync, "[%s] Unblock %s", id_str(), f.id_str());
228  m_queue.pop_front();
229  f.wakeup();
230  w->add(&f);
231  return true;
232  }
233 
234  bool unblockAll() noexcept
235  {
236  if(m_queue.empty())
237  return false;
238 
239  Worker* w = nullptr;
240  getContext(&w, nullptr);
241 
242  zth_dbg(sync, "[%s] Unblock all", id_str());
243 
244  while(!m_queue.empty()) {
245  Fiber& f = m_queue.front();
246  m_queue.pop_front();
247  f.wakeup();
248  w->add(&f);
249  }
250  return true;
251  }

414  void acquire(size_t count = 1)
415  {
416  size_t remaining = count;
417  while(remaining > 0) {
418  if(remaining <= m_count) {
419  m_count -= remaining;
420  if(m_count > 0)
421  // There might be another one waiting.
422  unblockFirst();
423  return;
424  } else {
425  remaining -= m_count;
426  m_count = 0;
427  block();
428  }
429  }
430  zth_dbg(sync, "[%s] Acquired %zu", id_str(), count);
431  }
432 
433  void release(size_t count = 1) noexcept
434  {
435  zth_assert(m_count + count >= m_count); // ...otherwise it wrapped around, which is
436  // probably not want you wanted...
437 
438  if(unlikely(m_count + count < m_count))
439  // wrapped around, saturate
440  m_count = std::numeric_limits<size_t>::max();
441  else
442  m_count += count;
443 
444  zth_dbg(sync, "[%s] Released %zu", id_str(), count);
445 
446  if(likely(m_count > 0))
447  unblockFirst();
448  }
449 
450  size_t value() const noexcept
451  {

453  }
