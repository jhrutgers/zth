98  class Synchronizer : public RefCounted, public UniqueID<Synchronizer> {
99  public:
100  Synchronizer(char const* name = "Synchronizer") : RefCounted(), UniqueID(Config::NamedSynchronizer ? name : NULL) {}
101  virtual ~Synchronizer() {
102  zth_dbg(sync, "[%s] Destruct", id_str());
103  zth_assert(m_queue.empty());
104  }
105 
106  protected:
107  void block() {
108  Worker* w;
109  Fiber* f;
110  getContext(&w, &f);
111 
112  zth_dbg(sync, "[%s] Block %s", id_str(), f->id_str());
113  w->release(*f);
114  m_queue.push_back(*f);
115  f->nap(Timestamp::null());
116  w->schedule();
117  }
118 
119  bool unblockFirst() {
120  if(m_queue.empty())
121  return false;
122 
123  Worker* w;
124  getContext(&w, NULL);
125 
126  Fiber& f = m_queue.front();
127  zth_dbg(sync, "[%s] Unblock %s", id_str(), f.id_str());
128  m_queue.pop_front();
129  f.wakeup();
130  w->add(&f);
131  return true;
132  }
133 
134  bool unblockAll() {
135  if(m_queue.empty())
136  return false;
137 
138  Worker* w;
139  getContext(&w, NULL);
140 
141  zth_dbg(sync, "[%s] Unblock all", id_str());
142 
143  while(!m_queue.empty()) {
144  Fiber& f = m_queue.front();
145  m_queue.pop_front();
146  f.wakeup();
147  w->add(&f);
148  }
149  return true;
150  }

193  class Semaphore : public Synchronizer {
194  public:
195  Semaphore(size_t init = 0, char const* name = "Semaphore") : Synchronizer(name), m_count(init) {}
196  virtual ~Semaphore() {}
197 
198  void acquire(size_t count = 1) {
199  size_t remaining = count;
200  while(remaining > 0) {
201  if(remaining <= m_count) {
202  m_count -= remaining;
203  if(m_count > 0)
204  // There might be another one waiting.
205  unblockFirst();
206  return;
207  } else {
208  remaining -= m_count;
209  m_count = 0;
210  block();
211  }
212  }
213  zth_dbg(sync, "[%s] Acquired %zu", id_str(), count);
214  }
215 
216  void release(size_t count = 1) {
217  zth_assert(m_count + count >= m_count); // ...otherwise it wrapped around, which is probably not want you wanted...
218 
219  if(unlikely(m_count + count < m_count))
220  // wrapped around, saturate
221  m_count = std::numeric_limits<size_t>::max();
222  else
223  m_count += count;
224 
225  zth_dbg(sync, "[%s] Released %zu", id_str(), count);
226 
227  if(likely(m_count > 0))
228  unblockFirst();
229  }
230 
231  size_t value() const { return m_count; }
232 
233  private:
234  size_t m_count;
235  };

259  zth_dbg(sync, "[%s] Signal", id_str());
260  if(!unblockFirst() && queue && m_signalled >= 0) {
261  if(m_signalled == 0 || queueEveryTime)
262  m_signalled++;
263  zth_assert(m_signalled > 0); // Otherwise, it wrapped around, which is probably not what you want.
264  }
