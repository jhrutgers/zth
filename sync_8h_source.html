163class Synchronizer : public RefCounted, public UniqueID<Synchronizer> {
164 ZTH_CLASS_NEW_DELETE(Synchronizer)
165public:
166 explicit Synchronizer(cow_string const& name = "Synchronizer")
167 : RefCounted()
168 , UniqueID(Config::NamedSynchronizer ? name.str() : string())
169 {}
170
171# if __cplusplus >= 201103L
172 explicit Synchronizer(cow_string&& name)
173 : RefCounted()
174 , UniqueID(Config::NamedSynchronizer ? std::move(name).str() : string())
175 {}
176# endif
177
178 virtual ~Synchronizer() override
179 {
180 zth_dbg(sync, "[%s] Destruct", id_str());
181 zth_assert(m_queue.empty());
182 }
183
184protected:
185 void block()
186 {
187 Worker* w = nullptr;
188 Fiber* f = nullptr;
189 getContext(&w, &f);
190
191 zth_dbg(sync, "[%s] Block %s", id_str(), f->id_str());
192 w->release(*f);
193 m_queue.push_back(*f);
194 f->nap(Timestamp::null());
195 w->schedule();
196 }
197
202 bool unblock(Fiber& f) noexcept
203 {
204 if(!m_queue.contains(f))
205 return false;
206
207 Worker* w = nullptr;
208 getContext(&w, nullptr);
209
210 zth_dbg(sync, "[%s] Unblock %s", id_str(), f.id_str());
211 m_queue.erase(f);
212 f.wakeup();
213 w->add(&f);
214 return true;
215 }
216
217 bool unblockFirst() noexcept
218 {
219 if(m_queue.empty())
220 return false;
221
222 Worker* w = nullptr;
223 getContext(&w, nullptr);
224
225 Fiber& f = m_queue.front();
226 zth_dbg(sync, "[%s] Unblock %s", id_str(), f.id_str());
227 m_queue.pop_front();
228 f.wakeup();
229 w->add(&f);
230 return true;
231 }
232
233 bool unblockAll() noexcept
234 {
235 if(m_queue.empty())
236 return false;
237
238 Worker* w = nullptr;
239 getContext(&w, nullptr);
240
241 zth_dbg(sync, "[%s] Unblock all", id_str());
242
243 while(!m_queue.empty()) {
244 Fiber& f = m_queue.front();
245 m_queue.pop_front();
246 f.wakeup();
247 w->add(&f);
248 }
249 return true;
250 }

288 friend class AlarmClock;
289
294 bool block(Timestamp const& timeout, Timestamp const& now = Timestamp::now())
295 {
296 if(timeout <= now)
297 // Immediate timeout.
298 return true;
299
300 return block_(timeout, now);
301 }
302
307 bool block(TimeInterval const& timeout, Timestamp const& now = Timestamp::now())
308 {
309 if(timeout.isNegative() || timeout.isNull())
310 // Immediate timeout.
311 return true;
312
313 return block_(now + timeout, now);
314 }
315
316private:
321 bool block_(Timestamp const& timeout, Timestamp const& now)
322 {
323 Worker* w;
324 Fiber* f;
325 getContext(&w, &f);
326
327 zth_dbg(sync, "[%s] Block %s with timeout", id_str(), f->id_str());
328 w->release(*f);
329 m_queue.push_back(*f);
330 f->nap(Timestamp::null());
331
332 AlarmClock a(*this, *f, timeout);
333 w->waiter().scheduleTask(a);
334 w->schedule(nullptr, now);
335
336 w->waiter().unscheduleTask(a);
337 return !a.rang();
338 }
339
340private:
341 List<Fiber> m_queue;
342};

446class Semaphore : public Synchronizer {
447 ZTH_CLASS_NEW_DELETE(Semaphore)
448public:
449 explicit Semaphore(size_t init = 0, cow_string const& name = "Semaphore")
450 : Synchronizer(name)
451 , m_count(init)
452 {}
453
454# if __cplusplus >= 201103L
455 Semaphore(size_t init, cow_string&& name)
456 : Synchronizer(std::move(name))
457 , m_count(init)
458 {}
459# endif
460
461 virtual ~Semaphore() override is_default
462
463 void acquire(size_t count = 1)
464 {
465 size_t remaining = count;
466 while(remaining > 0) {
467 if(remaining <= m_count) {
468 m_count -= remaining;
469 if(m_count > 0)
470 // There might be another one waiting.
471 unblockFirst();
472 return;
473 } else {
474 remaining -= m_count;
475 m_count = 0;
476 block();
477 }
478 }
479 zth_dbg(sync, "[%s] Acquired %zu", id_str(), count);
480 }
481
482 void release(size_t count = 1) noexcept
483 {
484 zth_assert(m_count + count >= m_count); // ...otherwise it wrapped around, which is
485 // probably not want you wanted...
486
487 if(unlikely(m_count + count < m_count))
488 // wrapped around, saturate
489 m_count = std::numeric_limits<size_t>::max();
490 else
491 m_count += count;
492
493 zth_dbg(sync, "[%s] Released %zu", id_str(), count);
494
495 if(likely(m_count > 0))
496 unblockFirst();
497 }
498
499 size_t value() const noexcept
500 {
501 return m_count;
502 }
503
504private:
505 size_t m_count;
506};

512class Signal : public Synchronizer {
513 ZTH_CLASS_NEW_DELETE(Signal)
514public:
515 explicit Signal(cow_string const& name = "Signal")
516 : Synchronizer(name)
517 , m_signalled()
518 {}
519
520# if __cplusplus >= 201103L
521 explicit Signal(cow_string&& name)
522 : Synchronizer(std::move(name))
523 , m_signalled()
524 {}
525# endif
526
527 virtual ~Signal() override is_default
528
529 void wait()
530 {
531 if(!m_signalled) {
532 block();
533 } else {
534 // Do a yield() here, as one might rely on the signal to block
535 // regularly when the signal is used in a loop (see daemon
536 // pattern).
537 yield();
538 }
539
540 if(m_signalled > 0)
541 m_signalled--;
542 }
543
544 bool wait(Timestamp const& timeout, Timestamp const& now = Timestamp::now())
545 {
546 if(!m_signalled) {
547 if(!block(timeout, now))
548 // Timeout.
549 return false;
550 } else
551 yield();
552
553 if(m_signalled > 0)
554 m_signalled--;
555
556 // Got signalled.
557 return true;
558 }
559
560 bool wait(TimeInterval const& timeout, Timestamp const& now = Timestamp::now())
561 {
562 return wait(now + timeout);
563 }
564
565 void signal(bool queue = true, bool queueEveryTime = false) noexcept
566 {
567 zth_dbg(sync, "[%s] Signal", id_str());
568 if(!unblockFirst() && queue && m_signalled >= 0) {
569 if(m_signalled == 0 || queueEveryTime)
570 m_signalled++;
571 zth_assert(m_signalled > 0); // Otherwise, it wrapped around, which is
572 // probably not what you want.
573 }
574 }
575
576 void signalAll(bool queue = true) noexcept
577 {
578 zth_dbg(sync, "[%s] Signal all", id_str());
579 unblockAll();
580 if(queue)
581 m_signalled = -1;
582 }
583
584 void reset() noexcept
585 {
586 m_signalled = 0;
587 }
588
589private:
590 int m_signalled;
