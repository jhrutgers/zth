53  }
54 
55  void unused() {
56  zth_assert(m_count > 0);

98  class Synchronizer : public RefCounted, public UniqueID<Synchronizer> {
99  public:
100  explicit Synchronizer(char const* name = "Synchronizer") : RefCounted(), UniqueID(Config::NamedSynchronizer ? name : NULL) {}
101  virtual ~Synchronizer() {
102  zth_dbg(sync, "[%s] Destruct", id_str());
103  zth_assert(m_queue.empty());
104  }
105 
106  protected:
107  void block() {
108  Worker* w;
109  Fiber* f;
110  getContext(&w, &f);
111 
112  zth_dbg(sync, "[%s] Block %s", id_str(), f->id_str());
113  w->release(*f);
114  m_queue.push_back(*f);
115  f->nap(Timestamp::null());
116  w->schedule();
117  }
118 
123  bool unblock(Fiber& f) {
124  if(!m_queue.contains(f))
125  return false;
126 
127  Worker* w;
128  getContext(&w, NULL);
129 
130  zth_dbg(sync, "[%s] Unblock %s", id_str(), f.id_str());
131  m_queue.erase(f);
132  f.wakeup();
133  w->add(&f);
134  return true;
135  }
136 
137  bool unblockFirst() {
138  if(m_queue.empty())
139  return false;
140 
141  Worker* w;
142  getContext(&w, NULL);
143 
144  Fiber& f = m_queue.front();
145  zth_dbg(sync, "[%s] Unblock %s", id_str(), f.id_str());
146  m_queue.pop_front();
147  f.wakeup();
148  w->add(&f);
149  return true;
150  }
151 
152  bool unblockAll() {
153  if(m_queue.empty())
154  return false;
155 
156  Worker* w;
157  getContext(&w, NULL);
158 
159  zth_dbg(sync, "[%s] Unblock all", id_str());
160 
161  while(!m_queue.empty()) {
162  Fiber& f = m_queue.front();
163  m_queue.pop_front();
164  f.wakeup();
165  w->add(&f);
166  }
167  return true;
168  }

205  bool block(Timestamp const& timeout, Timestamp const& now = Timestamp::now()) {
206  if(timeout <= now)
207  // Immediate timeout.
208  return true;
209 
210  return block_(timeout, now);
211  }
212 
217  bool block(TimeInterval const& timeout, Timestamp const& now = Timestamp::now()) {
218  if(timeout.isNegative() || timeout.isNull())
219  // Immediate timeout.
220  return true;
221 
222  return block_(now + timeout, now);
223  }
224 
225  private:
230  bool block_(Timestamp const& timeout, Timestamp const& now) {
231  Worker* w;
232  Fiber* f;
233  getContext(&w, &f);
234 
235  zth_dbg(sync, "[%s] Block %s with timeout", id_str(), f->id_str());
236  w->release(*f);
237  m_queue.push_back(*f);
238  f->nap(Timestamp::null());
239 
240  AlarmClock a(*this, *f, timeout);
241  w->waiter().scheduleTask(a);
242  w->schedule(NULL, now);
243 
244  w->waiter().unscheduleTask(a);
245  return !a.rang();
246  }
247 
248  private:
249  List<Fiber> m_queue;
250  };

289  class Semaphore : public Synchronizer {
290  public:
291  Semaphore(size_t init = 0, char const* name = "Semaphore") : Synchronizer(name), m_count(init) {}
292  virtual ~Semaphore() {}
293 
294  void acquire(size_t count = 1) {
295  size_t remaining = count;
296  while(remaining > 0) {
297  if(remaining <= m_count) {
298  m_count -= remaining;
299  if(m_count > 0)
300  // There might be another one waiting.
301  unblockFirst();
302  return;
303  } else {
304  remaining -= m_count;
305  m_count = 0;
306  block();
307  }
308  }
309  zth_dbg(sync, "[%s] Acquired %zu", id_str(), count);
310  }
311 
312  void release(size_t count = 1) {
313  zth_assert(m_count + count >= m_count); // ...otherwise it wrapped around, which is probably not want you wanted...
314 
315  if(unlikely(m_count + count < m_count))
316  // wrapped around, saturate
317  m_count = std::numeric_limits<size_t>::max();
318  else
319  m_count += count;
320 
321  zth_dbg(sync, "[%s] Released %zu", id_str(), count);
322 
323  if(likely(m_count > 0))
324  unblockFirst();
325  }
326 
327  size_t value() const { return m_count; }
328 
329  private:
330  size_t m_count;
331  };

336  class Signal : public Synchronizer {
337  public:
338  explicit Signal(char const* name = "Signal") : Synchronizer(name) , m_signalled() {}
339  virtual ~Signal() {}
340 
341  void wait() {
342  if(!m_signalled)
343  block();
344  else
345  // Do a yield() here, as one might rely on the signal to block
346  // regularly when the signal is used in a loop (see daemon
347  // pattern).
348  yield();
349 
350  if(m_signalled > 0)
351  m_signalled--;
352  }
353 
354  bool wait(Timestamp const& timeout, Timestamp const& now = Timestamp::now()) {
355  if(!m_signalled) {
356  if(!block(timeout, now))
357  // Timeout.
358  return false;
359  } else
360  yield();
361 
362  if(m_signalled > 0)
363  m_signalled--;
364 
365  // Got signalled.
366  return true;
367  }
368 
369  bool wait(TimeInterval const& timeout, Timestamp const& now = Timestamp::now()) {
370  return wait(now + timeout);
371  }

373  void signal(bool queue = true, bool queueEveryTime = false) {
374  zth_dbg(sync, "[%s] Signal", id_str());
375  if(!unblockFirst() && queue && m_signalled >= 0) {
376  if(m_signalled == 0 || queueEveryTime)
377  m_signalled++;
378  zth_assert(m_signalled > 0); // Otherwise, it wrapped around, which is probably not what you want.
379  }
380  }

383  zth_dbg(sync, "[%s] Signal all", id_str());
384  unblockAll();
385  if(queue)
386  m_signalled = -1;
387  }
388 
389  void reset() {
390  m_signalled = 0;
391  }

394  int m_signalled;
