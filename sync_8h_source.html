99  class Synchronizer : public RefCounted, public UniqueID<Synchronizer> {
100  ZTH_CLASS_NEW_DELETE(Synchronizer)
101  public:
102  explicit Synchronizer(char const* name = "Synchronizer") : RefCounted(), UniqueID(Config::NamedSynchronizer ? name : NULL) {}
103  virtual ~Synchronizer() {
104  zth_dbg(sync, "[%s] Destruct", id_str());
105  zth_assert(m_queue.empty());
106  }
107 
108  protected:
109  void block() {
110  Worker* w;
111  Fiber* f;
112  getContext(&w, &f);
113 
114  zth_dbg(sync, "[%s] Block %s", id_str(), f->id_str());
115  w->release(*f);
116  m_queue.push_back(*f);
117  f->nap(Timestamp::null());
118  w->schedule();
119  }
120 
125  bool unblock(Fiber& f) {
126  if(!m_queue.contains(f))
127  return false;
128 
129  Worker* w;
130  getContext(&w, NULL);
131 
132  zth_dbg(sync, "[%s] Unblock %s", id_str(), f.id_str());
133  m_queue.erase(f);
134  f.wakeup();
135  w->add(&f);
136  return true;
137  }
138 
139  bool unblockFirst() {
140  if(m_queue.empty())
141  return false;
142 
143  Worker* w;
144  getContext(&w, NULL);
145 
146  Fiber& f = m_queue.front();
147  zth_dbg(sync, "[%s] Unblock %s", id_str(), f.id_str());
148  m_queue.pop_front();
149  f.wakeup();
150  w->add(&f);
151  return true;
152  }
153 
154  bool unblockAll() {
155  if(m_queue.empty())
156  return false;
157 
158  Worker* w;
159  getContext(&w, NULL);
160 
161  zth_dbg(sync, "[%s] Unblock all", id_str());
162 
163  while(!m_queue.empty()) {
164  Fiber& f = m_queue.front();
165  m_queue.pop_front();
166  f.wakeup();
167  w->add(&f);
168  }
169  return true;
170  }

207  bool block(Timestamp const& timeout, Timestamp const& now = Timestamp::now()) {
208  if(timeout <= now)
209  // Immediate timeout.
210  return true;
211 
212  return block_(timeout, now);
213  }
214 
219  bool block(TimeInterval const& timeout, Timestamp const& now = Timestamp::now()) {
220  if(timeout.isNegative() || timeout.isNull())
221  // Immediate timeout.
222  return true;
223 
224  return block_(now + timeout, now);
225  }
226 
227  private:
232  bool block_(Timestamp const& timeout, Timestamp const& now) {
233  Worker* w;
234  Fiber* f;
235  getContext(&w, &f);
236 
237  zth_dbg(sync, "[%s] Block %s with timeout", id_str(), f->id_str());
238  w->release(*f);
239  m_queue.push_back(*f);
240  f->nap(Timestamp::null());
241 
242  AlarmClock a(*this, *f, timeout);
243  w->waiter().scheduleTask(a);
244  w->schedule(NULL, now);
245 
246  w->waiter().unscheduleTask(a);
247  return !a.rang();
248  }
249 
250  private:
251  List<Fiber> m_queue;
252  };

292  class Semaphore : public Synchronizer {
293  ZTH_CLASS_NEW_DELETE(Semaphore)
294  public:
295  Semaphore(size_t init = 0, char const* name = "Semaphore") : Synchronizer(name), m_count(init) {}
296  virtual ~Semaphore() {}
297 
298  void acquire(size_t count = 1) {
299  size_t remaining = count;
300  while(remaining > 0) {
301  if(remaining <= m_count) {
302  m_count -= remaining;
303  if(m_count > 0)
304  // There might be another one waiting.
305  unblockFirst();
306  return;
307  } else {
308  remaining -= m_count;
309  m_count = 0;
310  block();
311  }
312  }
313  zth_dbg(sync, "[%s] Acquired %zu", id_str(), count);
314  }
315 
316  void release(size_t count = 1) {
317  zth_assert(m_count + count >= m_count); // ...otherwise it wrapped around, which is probably not want you wanted...
318 
319  if(unlikely(m_count + count < m_count))
320  // wrapped around, saturate
321  m_count = std::numeric_limits<size_t>::max();
322  else
323  m_count += count;
324 
325  zth_dbg(sync, "[%s] Released %zu", id_str(), count);
326 
327  if(likely(m_count > 0))
328  unblockFirst();
329  }
330 
331  size_t value() const { return m_count; }
332 
333  private:
334  size_t m_count;
335  };

340  class Signal : public Synchronizer {
341  ZTH_CLASS_NEW_DELETE(Signal)
342  public:
343  explicit Signal(char const* name = "Signal") : Synchronizer(name) , m_signalled() {}
344  virtual ~Signal() {}
345 
346  void wait() {
347  if(!m_signalled)
348  block();
349  else
350  // Do a yield() here, as one might rely on the signal to block
351  // regularly when the signal is used in a loop (see daemon
352  // pattern).
353  yield();
354 
355  if(m_signalled > 0)
356  m_signalled--;
357  }
358 
359  bool wait(Timestamp const& timeout, Timestamp const& now = Timestamp::now()) {
360  if(!m_signalled) {
361  if(!block(timeout, now))
362  // Timeout.
363  return false;
364  } else
365  yield();
366 
367  if(m_signalled > 0)
368  m_signalled--;
369 
370  // Got signalled.
371  return true;
372  }
373 
374  bool wait(TimeInterval const& timeout, Timestamp const& now = Timestamp::now()) {
375  return wait(now + timeout);
376  }

378  void signal(bool queue = true, bool queueEveryTime = false) {
379  zth_dbg(sync, "[%s] Signal", id_str());
380  if(!unblockFirst() && queue && m_signalled >= 0) {
381  if(m_signalled == 0 || queueEveryTime)
382  m_signalled++;
383  zth_assert(m_signalled > 0); // Otherwise, it wrapped around, which is probably not what you want.
384  }
385  }

387  void signalAll(bool queue = true) {
388  zth_dbg(sync, "[%s] Signal all", id_str());
389  unblockAll();
390  if(queue)
391  m_signalled = -1;
392  }
393 
394  void reset() {
395  m_signalled = 0;
396  }

399  int m_signalled;
