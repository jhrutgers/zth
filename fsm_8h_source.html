211  std::map<typename Fsm::State,FsmDescription<Fsm> const*> stateAddr;
212  FsmDescription<Fsm>* p = m_description;

214  // Register this state

220  break;

222  // Go to end of guard list
223  while((p++)->guard.valid());

227  p = m_description;
228  while(true) {
229  if(!p->guard.valid())
230  // End of description
231  break;
232 
233  // Patch this state's guard list
234  while((p++)->guard.valid()) {
235  zth_assert(stateAddr.count(p->state) == 1);
236  p->stateAddr = stateAddr[p->state];
237  }
238  }

240  m_compiled = true;
241  }

491  void run() {
492  TimeInterval wait = eval();
493  if(!m_state->guard.valid()) {
494  done:
495  zth_dbg(fsm, "[%s] Final state %s", id_str(), str(state()).c_str());

500  do {

503  } else {

506  unscheduleTask(wakeup);
507  }

524  TimeInterval evalOnce() {
525  TimeInterval wait = TimeInterval::infinity();
526  StateAddr p = m_state;
527  while(p->guard.valid()) {
528  TimeInterval ti((p++)->guard(static_cast<FsmImpl&>(*this)));
529  if(ti.hasPassed()) {
530  setState(p->stateAddr);
531  wait = ti; // NVRO
532  return wait; // Return any passed TimeInterval, which indicates that a transition was taken.
533  } else if(ti < wait) {
534  // Save shortest wait interval.
535  wait = ti;
536  }
537  }
538  zth_dbg(fsm, "[%s] State %s blocked for %s", id_str(), str(state()).c_str(), wait.str().c_str());
