249  FsmDescription<Fsm> const* description() const
250  {

255 protected:
256  void compile() const
257  {
258  if(likely(m_compiled))
259  return;
260 
261  typename map_type<typename Fsm::State, FsmDescription<Fsm> const*>::type stateAddr;
262  FsmDescription<Fsm>* p = m_description;
263  while(true) {
264  // Register this state
265  zth_assert(stateAddr.count(p->state) == 0);
266  stateAddr[p->state] = const_cast<FsmDescription<Fsm> const*>(p);
267 
268  if(!p->guard.valid())
269  // End of description
270  break;
271 
272  // Go to end of guard list
273  while((p++)->guard.valid())
274  ;
275  }
276 
277  // Patch State references
278  p = m_description;
279  while(true) {
280  if(!p->guard.valid())
281  // End of description
282  break;
283 
284  // Patch this state's guard list
285  while((p++)->guard.valid()) {
286  zth_assert(stateAddr.count(p->state) == 1);
287  p->stateAddr = stateAddr[p->state];
288  }
289  }
290 
291  m_compiled = true;
292  }
293 
294 private:
295  FsmDescription<Fsm>* const m_description;
296  mutable bool m_compiled;
297 };

573  {
574  bool didCallback = false;
575  TimeInterval wait = TimeInterval::null();
576 
577  switch(m_evalState) {
578  case evalCompile:
579  m_compiledDescription = Compiler(m_description).description();
580  if(false) {
581  ZTH_FALLTHROUGH
582  case evalInit:
583  m_compiledDescription = (StateAddr)m_compiler->description();
584  }
585  ZTH_FALLTHROUGH

592  m_entry = true;
593  callback();
594  m_entry = false;

596  ZTH_FALLTHROUGH

654  {
655  TimeInterval wait = TimeInterval::infinity();
656  StateAddr p = m_state;
657  while(p->guard.valid()) {
658  TimeInterval ti((p++)->guard(static_cast<FsmImpl&>(*this)));
659  if(ti.hasPassed()) {
