41 int clock_gettime(int clk_id, struct timespec* res) {
42  if(unlikely(!res))
43  return EFAULT;
44 
45  zth_assert(clk_id == CLOCK_MONOTONIC);
46  uint64_t c = (mach_absolute_time() - mach_clock_start);
47  uint64_t ns;
48  if(unlikely(clock_info.numer != clock_info.denom)) // For all recent Mac OSX versions, mach_absolute_time() is in ns.
49  {
50  uint64_t chigh = (c >> 32) * clock_info.numer;
51  uint64_t chighrem = ((chigh % clock_info.denom) << 32) / clock_info.denom;
52  chigh /= clock_info.denom;
53  uint64_t clow = (c & (((uint64_t)1 << 32) - 1)) * clock_info.numer / clock_info.denom;
54  clow += chighrem;
55  ns = (chigh << 32) + clow; // 64-bit ns gives us more than 500 y before wrap-around.
56  } else {
57  ns = c;
58  }
59 
60  // Split in sec + nsec
61  res->tv_nsec = (long)(ns % zth::TimeInterval::BILLION);
62  res->tv_sec = (time_t)(ns / zth::TimeInterval::BILLION);
63  return 0;
64 }

67 int clock_nanosleep(int clk_id, int flags, struct timespec const* request, struct timespec* remain) {
68  if(unlikely(!request))
69  return EFAULT;
70  if(unlikely(clk_id != CLOCK_MONOTONIC || flags != TIMER_ABSTIME))
71  return EINVAL;
72 
73  struct timespec now;
74  int res = clock_gettime(CLOCK_MONOTONIC, &now);
75  if(unlikely(res))
76  return res;
77 
78  if(now.tv_sec > request->tv_sec)
79  // No need to sleep.
80  return 0;
81 
82  time_t sec = request->tv_sec - now.tv_sec;
83  if(sec == 0 && request->tv_nsec <= now.tv_nsec) {
84  // No need to sleep.
85  return 0;
86  } else if(sec < std::numeric_limits<useconds_t>::max() / 1000000 - 1) {
87  useconds_t us = (useconds_t)(sec * 1000000);
88  us += (useconds_t)((request->tv_nsec - now.tv_nsec) / 1000);
89  if(likely(!usleep(us)))
90  return 0;
91  } else {
92  // Overflow, sleep (almost) infinitely.
93  if(likely(!usleep(std::numeric_limits<useconds_t>::max())))
94  return 0;
95  }
96 
97  if(remain) {
98  struct timespec intr;
99  if((res = clock_gettime(CLOCK_MONOTONIC, &intr)))
100  return res;
101 
102  remain->tv_sec = intr.tv_sec - now.tv_sec;
103  remain->tv_nsec = intr.tv_nsec - now.tv_nsec;
104  if(remain->tv_nsec < 0) {
105  remain->tv_nsec += 1000000000L;
106  remain->tv_sec--;
107  }
108  }
109 
110  return EINTR;
111 }
112 #endif
