46 void Waiter::wait(TimedWaitable& w)
47 {
48  Fiber* fiber = m_worker.currentFiber();
49  if(unlikely(!fiber || fiber->state() != Fiber::Running))
50  return;
51 
52  Timestamp now = Timestamp::now();
53  if(unlikely(w.poll(now))) {
54  yield(NULL, false, now);
55  return;
56  }
57 
58  w.setFiber(*fiber);
59  fiber->nap(w.timeout());
60  m_worker.release(*fiber);
61 
62  m_waiting.insert(w);
63  if(this->fiber())
64  m_worker.resume(*this->fiber());
65 
66  m_worker.schedule();
67 }

140 void Waiter::setPoller(PollerServerBase* p)
141 {
142  if(!m_poller) {
143  // No poller set, just use the given one (which may be nullptr).
144  m_poller = p;
145  } else if(p) {
146  // Replace poller by another one.
147  m_poller->migrateTo(*p);
148  m_poller = p;
149  } else {
150  // Replace poller by default one.
151  if(!m_defaultPoller && !m_poller->empty()) {
152  // ...but it doesn't exist yet and we need it.
153  m_defaultPoller = new_alloc<DefaultPollerServer>();
154  }
155 
156  if(m_defaultPoller)
157  m_poller->migrateTo(*m_defaultPoller);
158 
159  m_poller = m_defaultPoller;

163 void Waiter::entry()
164 {
165  zth_assert(&currentWorker() == &m_worker);
166  fiber()->setName(format("zth::Waiter of %s", m_worker.id_str()));
167 
168  while(true) {
169  Timestamp now = Timestamp::now();
170  m_worker.load().stop(now);
171 
172  while(!m_waiting.empty() && m_waiting.front().timeout() < now) {
173  TimedWaitable& w = m_waiting.front();
174  m_waiting.erase(w);
175  if(w.poll(now)) {
176  if(w.hasFiber()) {
177  w.fiber().wakeup();
178  m_worker.add(&w.fiber());
179  }
180  } else {
181  // Reinsert, as the timeout() might have changed (and therefore the position in the list).
182  m_waiting.insert(w);
183 
184  // Update administration, although that does not influence the actual sleep time.
185  // It is mostly handy for debugging while printing fiber information.
186  if(Config::Debug && w.hasFiber())
187  w.fiber().nap(w.timeout());
188  }
189  }
190 
191  bool doRealSleep = false;
192 
193  m_worker.load().start(now);
194 
195  if(m_waiting.empty() && !polling()) {
196  // No fiber is waiting. suspend() till anyone is going to nap().
197  zth_dbg(waiter, "[%s] No sleeping fibers anymore; suspend", id_str());
198  m_worker.suspend(*fiber());
199  } else if(!m_worker.schedule()) {
200  // When true, we were not rescheduled, which means that we are the only runnable fiber.
201  // Do a real sleep, until something interesting happens in the system.
202  doRealSleep = true;
203  m_worker.load().stop(now);
204  }
205 
206  Timestamp const* end = nullptr;
207  if(!m_waiting.empty())
208  end = &m_waiting.front().timeout();
209  if(!m_worker.runEnd().isNull() && (!end || *end > m_worker.runEnd()))
210  end = &m_worker.runEnd();
211 
212  if(polling()) {
213  int timeout_ms = 0;
214  if(doRealSleep) {
215  if(!end) {
216  // Infinite sleep.
217  timeout_ms = -1;
218  zth_dbg(waiter, "[%s] Out of other work than doing poll()", id_str());
219  } else {
220  TimeInterval dt = *end - Timestamp::now();
221  zth_dbg(waiter, "[%s] Out of other work than doing poll(); timeout is %s", id_str(), dt.str().c_str());
222  timeout_ms = std::max<int>(0, (int)(dt.s<float>() * 1000.0f));
223  }
224 
225  perf_mark("blocking poll()");
226  perf_event(PerfEvent<>(*fiber(), Fiber::Waiting));
227  }
228 
229  int res = poller().poll(timeout_ms);
230 
231  if(doRealSleep) {
232  perf_event(PerfEvent<>(*fiber(), fiber()->state()));
233  perf_mark("wakeup");
234  }
235 
236  if(res && res != EAGAIN)
237  zth_dbg(waiter, "[%s] poll() failed; %s", id_str(), err(res).c_str());
238  } else if(doRealSleep) {
239  zth_dbg(waiter, "[%s] Out of work; suspend thread, while waiting for %s", id_str(), m_waiting.front().str().c_str());
240  perf_mark("idle system; sleep");
241  perf_event(PerfEvent<>(*fiber(), Fiber::Waiting));
242  clock_nanosleep(CLOCK_MONOTONIC, TIMER_ABSTIME, &end->ts(), NULL);
243  perf_event(PerfEvent<>(*fiber(), fiber()->state()));
244  perf_mark("wakeup");
245  }
246 
247  sigchld_check();
248  }

255 bool PeriodicWakeUp::nap(Timestamp const& reference, Timestamp const& now) {
256  m_t = reference + interval();
257  if(likely(m_t > now)) {
258  // Proper sleep till next deadline.
259  zth::nap(m_t);
260  return true;
261  } else if(m_t + interval() > now) {
262  // Deadline has just passed. Don't sleep and try to catch up.
263  yield();
264  return false;
265  } else {
266  // Way passed deadline. Don't sleep and skip a few cycles.
267  m_t = now;
268  yield();
269  return false;
270  }
271 }
