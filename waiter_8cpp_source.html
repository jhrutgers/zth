30 void Waiter::wait(TimedWaitable& w) {
31  Fiber* fiber = m_worker.currentFiber();
32  if(unlikely(!fiber || fiber->state() != Fiber::Running))
33  return;
34 
35  Timestamp now = Timestamp::now();
36  if(unlikely(w.poll(now))) {
37  yield(NULL, false, now);
38  return;
39  }
40 
41  w.setFiber(*fiber);
42  fiber->nap(w.timeout());
43  m_worker.release(*fiber);
44 
45  m_waiting.insert(w);
46  if(this->fiber())
47  m_worker.resume(*this->fiber());
48 
49  m_worker.schedule();
50 }

74  if(!Config::EnableAssert && (!zth_config(EnableDebugPrint) || !Config::Print_list))
75  return;
76 
77  zth_dbg(list, "[%s] Waiter poll() list:", id_str());

80  zth_dbg(list, "[%s] %s", id_str(), itw->str().c_str());

82 
83  for(size_t i = 0; i < (size_t)itw->nfds(); offset++, i++) {
84  std::string fd;
85 # ifdef ZTH_HAVE_LIBZMQ
86  if(m_fdPollList[i].socket)
87  fd = format("0x%" PRIxPTR, (uintptr_t)m_fdPollList[i].socket);
88  else
89 # endif
90  format("%lld", (long long)m_fdPollList[i].fd);
91  zth_dbg(list, "[%s] %s: events 0x%04hx", id_str(), fd.c_str(), m_fdPollList[i].events);

97 int Waiter::waitFd(AwaitFd& w) {
98  Fiber* fiber = m_worker.currentFiber();
99  if(unlikely(!fiber || fiber->state() != Fiber::Running))
100  return EAGAIN;
101 
102  w.setFiber(*fiber);
103 
104  // Add our set of fds to the Waiter's administration.
105  m_fdList.push_back(w);
106  m_fdPollList.reserve(m_fdPollList.size() + w.nfds());
107  for(int i = 0; i < w.nfds(); i++)
108  m_fdPollList.push_back(w.fds()[i]);
109 
110  checkFdList();
111 
112  // Put ourselves to sleep.
113  fiber->nap();
114  m_worker.release(*fiber);
115 
116  // Switch to Waiter
117  if(this->fiber())
118  m_worker.resume(*this->fiber());
119 
120  m_worker.schedule();
121 
122  // Got back, check which fds were triggered.
123  zth_assert(w.finished());
124 
125  size_t offset = 0;
126  int res = 0;
127  for(decltype(m_fdList.begin()) itw = m_fdList.begin(); itw != m_fdList.end(); ++itw, offset += itw->nfds())
128  if(&*itw == &w) {
129  // This is us
130  if(!w.error()) {
131  for(size_t i = 0; i < (size_t)w.nfds(); i++) {
132  zth_pollfd_t& f = w.fds()[i] = m_fdPollList[offset + i];
133  if(f.revents) {
134  zth_dbg(waiter, "[%s] poll(%lld)'s revents: 0x%04hx", id_str(), (long long)f.fd, f.events);
135  res++;
136  }
137  }
138  }
139 
140  m_fdList.erase(w);
141  m_fdPollList.erase(m_fdPollList.begin() + offset, m_fdPollList.begin() + offset + w.nfds());
142  break;
143  }
144 
145  checkFdList();
146 
147  if(w.error())
148  return w.error();
149 
150  w.setResult(res);
151  return 0;
152 }

155 void Waiter::entry() {
156  zth_assert(&currentWorker() == &m_worker);
157  fiber()->setName(format("zth::Waiter of %s", m_worker.id_str()));
158 
159  while(true) {
160  Timestamp now = Timestamp::now();
161  m_worker.load().stop(now);
162 
163  while(!m_waiting.empty() && m_waiting.front().timeout() < now) {
164  TimedWaitable& w = m_waiting.front();
165  m_waiting.erase(w);
166  if(w.poll(now)) {
167  if(w.hasFiber()) {
168  w.fiber().wakeup();
169  m_worker.add(&w.fiber());
170  }
171  } else {
172  // Reinsert, as the timeout() might have changed (and therefore the position in the list).
173  m_waiting.insert(w);
174  }
175  }
176 
177  bool doRealSleep = false;
178 
179  m_worker.load().start(now);
180 
181  if(m_waiting.empty()
182 #ifdef ZTH_HAVE_POLLER
183  && m_fdPollList.empty()
184 #endif
185  ) {
186  // No fiber is waiting. suspend() till anyone is going to nap().
187  zth_dbg(waiter, "[%s] No sleeping fibers anymore; suspend", id_str());
188  m_worker.suspend(*fiber());
189  } else if(!m_worker.schedule()) {
190  // When true, we were not rescheduled, which means that we are the only runnable fiber.
191  // Do a real sleep, until something interesting happens in the system.
192  doRealSleep = true;
193  m_worker.load().stop(now);
194  }
195 
196 #ifdef ZTH_HAVE_POLLER
197  if(!m_fdPollList.empty()) {
198  Timestamp const* pollTimeout = NULL;
199  if(doRealSleep) {
200  if(!m_worker.runEnd().isNull() && (!pollTimeout || *pollTimeout > m_worker.runEnd()))
201  pollTimeout = &m_worker.runEnd();
202  for(decltype(m_fdList.begin()) it = m_fdList.begin(); it != m_fdList.end(); ++it)
203  if(!it->timeout().isNull() && (!pollTimeout || *pollTimeout > it->timeout()))
204  pollTimeout = &it->timeout();
205  if(!m_waiting.empty() && (!pollTimeout || *pollTimeout > m_waiting.front().timeout()))
206  pollTimeout = &m_waiting.front().timeout();
207  }
208 
209  int timeout_ms = 0;
210  if(doRealSleep) {
211  if(!pollTimeout) {
212  // Infinite sleep.
213  timeout_ms = -1;
214  zth_dbg(waiter, "[%s] Out of other work than doing poll()", id_str());
215  } else {
216  TimeInterval dt = *pollTimeout - Timestamp::now();
217  zth_dbg(waiter, "[%s] Out of other work than doing poll(); timeout is %s", id_str(), dt.str().c_str());
218  timeout_ms = std::max<int>(0, (int)(dt.s() * 1000.0));
219  }
220  }
221 
222  if(doRealSleep) {
223  perf_mark("blocking poll()");
224  perf_event(PerfEvent<>(*fiber(), Fiber::Waiting));
225  }
226 
227 #ifdef ZTH_HAVE_LIBZMQ
228  int res = ::zmq_poll(&m_fdPollList[0], (int)m_fdPollList.size(), timeout_ms);
229  int error = res == -1 ? zmq_errno() : 0;
230 #else
231  int res = ::poll(&m_fdPollList[0], (nfds_t)m_fdPollList.size(), timeout_ms);
232  int error = res == -1 ? errno : 0;
233 #endif
234 
235  if(doRealSleep) {
236  perf_event(PerfEvent<>(*fiber(), fiber()->state()));
237  perf_mark("wakeup");
238  }
239 
240  if(res == -1) {
241  zth_dbg(waiter, "[%s] poll() failed; %s", id_str(), err(error).c_str());
242  for(decltype(m_fdList.begin()) it = m_fdList.begin(); it != m_fdList.end(); ++it) {
243  zth_dbg(waiter, "[%s] %s got error; wakeup", id_str(), it->str().c_str());
244  it->setResult(-1, error);
245  it->fiber().wakeup();
246  m_worker.add(&it->fiber());
247  }
248  } else if(res > 0) {
249  size_t offset = 0;
250  for(decltype(m_fdList.begin()) it = m_fdList.begin(); res > 0 && it != m_fdList.end(); offset += it->nfds(), ++it) {
251  bool wakeup = false;
252  zth_assert(m_fdPollList.size() >= offset + it->nfds());
253  for(size_t i = 0; res > 0 && i < (size_t)it->nfds(); i++) {
254  if(m_fdPollList[offset + i].revents) {
255  wakeup = true;
256  res--;
257  }
258  }
259 
260  if(wakeup) {
261  zth_dbg(waiter, "[%s] %s got ready; wakeup", id_str(), it->str().c_str());
262  it->setResult(0);
263  it->fiber().wakeup();
264  m_worker.add(&it->fiber());
265  }
266  }
267  }
268  } else
269 #endif
270  if(doRealSleep) {
271  zth_dbg(waiter, "[%s] Out of work; suspend thread, while waiting for %s", id_str(), m_waiting.front().str().c_str());
272  Timestamp const* end = &m_waiting.front().timeout();
273  if(!m_worker.runEnd().isNull () && *end > m_worker.runEnd())
274  end = &m_worker.runEnd();
275  perf_mark("idle system; sleep");
276  perf_event(PerfEvent<>(*fiber(), Fiber::Waiting));
277  clock_nanosleep(CLOCK_MONOTONIC, TIMER_ABSTIME, &end->ts(), NULL);
278  perf_event(PerfEvent<>(*fiber(), fiber()->state()));
279  perf_mark("wakeup");
280  }
281 
282  sigchld_check();
283  }

290 bool PeriodicWakeUp::nap(Timestamp const& reference, Timestamp const& now) {
291  m_t = reference + interval();
292  if(likely(m_t > now)) {
293  // Proper sleep till next deadline.
294  zth::nap(m_t);
295  return true;
296  } else if(m_t + interval() > now) {
297  // Deadline has just passed. Don't sleep and try to catch up.
298  yield();
299  return false;
300  } else {
301  // Way passed deadline. Don't sleep and skip a few cycles.
302  m_t = now;
303  yield();
304  return false;
305  }
306 }
