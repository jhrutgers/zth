43 void Waiter::wait(TimedWaitable& w)
44 {
45  Fiber* fiber = m_worker.currentFiber();
46  if(unlikely(!fiber || fiber->state() != Fiber::Running))
47  return;
48 
49  Timestamp now = Timestamp::now();
50  if(unlikely(w.poll(now))) {
51  yield(NULL, false, now);
52  return;
53  }
54 
55  w.setFiber(*fiber);
56  fiber->nap(w.timeout());
57  m_worker.release(*fiber);
58 
59  m_waiting.insert(w);
60  if(this->fiber())
61  m_worker.resume(*this->fiber());
62 
63  m_worker.schedule();
64 }

137 void Waiter::setPoller(PollerServerBase* p)
138 {
139  if(!m_poller) {
140  // No poller set, just use the given one (which may be nullptr).
141  m_poller = p;
142  } else if(p) {
143  // Replace poller by another one.
144  m_poller->migrateTo(*p);
145  m_poller = p;
146  } else {
147  // Replace poller by default one.
148  if(!m_defaultPoller && !m_poller->empty()) {
149  // ...but it doesn't exist yet and we need it.
150  m_defaultPoller = new DefaultPollerServer();
151  }
152 
153  if(m_defaultPoller)
154  m_poller->migrateTo(*m_defaultPoller);
155 
156  m_poller = m_defaultPoller;

160 void Waiter::entry()
161 {
162  zth_assert(&currentWorker() == &m_worker);
163  fiber()->setName(format("zth::Waiter of %s", m_worker.id_str()));
164 
165  while(true) {
166  Timestamp now = Timestamp::now();
167  m_worker.load().stop(now);
168 
169  while(!m_waiting.empty() && m_waiting.front().timeout() < now) {
170  TimedWaitable& w = m_waiting.front();
171  m_waiting.erase(w);
172  if(w.poll(now)) {
173  if(w.hasFiber()) {
174  w.fiber().wakeup();
175  m_worker.add(&w.fiber());
176  }
177  } else {
178  // Reinsert, as the timeout() might have changed (and therefore the position in the list).
179  m_waiting.insert(w);
180 
181  // Update administration, although that does not influence the actual sleep time.
182  // It is mostly handy for debugging while printing fiber information.
183  if(Config::Debug && w.hasFiber())
184  w.fiber().nap(w.timeout());
185  }
186  }
187 
188  bool doRealSleep = false;
189 
190  m_worker.load().start(now);
191 
192  if(m_waiting.empty() && !polling()) {
193  // No fiber is waiting. suspend() till anyone is going to nap().
194  zth_dbg(waiter, "[%s] No sleeping fibers anymore; suspend", id_str());
195  m_worker.suspend(*fiber());
196  } else if(!m_worker.schedule()) {
197  // When true, we were not rescheduled, which means that we are the only runnable fiber.
198  // Do a real sleep, until something interesting happens in the system.
199  doRealSleep = true;
200  m_worker.load().stop(now);
201  }
202 
203  Timestamp const* end = nullptr;
204  if(!m_waiting.empty())
205  end = &m_waiting.front().timeout();
206  if(!m_worker.runEnd().isNull() && (!end || *end > m_worker.runEnd()))
207  end = &m_worker.runEnd();
208 
209  if(polling()) {
210  int timeout_ms = 0;
211  if(doRealSleep) {
212  if(!end) {
213  // Infinite sleep.
214  timeout_ms = -1;
215  zth_dbg(waiter, "[%s] Out of other work than doing poll()", id_str());
216  } else {
217  TimeInterval dt = *end - Timestamp::now();
218  zth_dbg(waiter, "[%s] Out of other work than doing poll(); timeout is %s", id_str(), dt.str().c_str());
219  timeout_ms = std::max<int>(0, (int)(dt.s<float>() * 1000.0f));
220  }
221 
222  perf_mark("blocking poll()");
223  perf_event(PerfEvent<>(*fiber(), Fiber::Waiting));
224  }
225 
226  int res = poller().poll(timeout_ms);
227 
228  if(doRealSleep) {
229  perf_event(PerfEvent<>(*fiber(), fiber()->state()));
230  perf_mark("wakeup");
231  }
232 
233  if(res && res != EAGAIN)
234  zth_dbg(waiter, "[%s] poll() failed; %s", id_str(), err(res).c_str());
235  } else if(doRealSleep) {
236  zth_dbg(waiter, "[%s] Out of work; suspend thread, while waiting for %s", id_str(), m_waiting.front().str().c_str());
237  perf_mark("idle system; sleep");
238  perf_event(PerfEvent<>(*fiber(), Fiber::Waiting));
239  clock_nanosleep(CLOCK_MONOTONIC, TIMER_ABSTIME, &end->ts(), NULL);
240  perf_event(PerfEvent<>(*fiber(), fiber()->state()));
241  perf_mark("wakeup");
242  }
243 
244  sigchld_check();
245  }

252 bool PeriodicWakeUp::nap(Timestamp const& reference, Timestamp const& now) {
253  m_t = reference + interval();
254  if(likely(m_t > now)) {
255  // Proper sleep till next deadline.
256  zth::nap(m_t);
257  return true;
258  } else if(m_t + interval() > now) {
259  // Deadline has just passed. Don't sleep and try to catch up.
260  yield();
261  return false;
262  } else {
263  // Way passed deadline. Don't sleep and skip a few cycles.
264  m_t = now;
265  yield();
266  return false;
267  }
268 }
