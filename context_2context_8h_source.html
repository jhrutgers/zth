159 int create() noexcept
160 {
161 int res = 0;
162
163 m_stack = Stack(attr().stackSize);
164 m_stackUsable = Stack();
165
166 if(attr().stackSize == 0)
167 // No stack requested.
168 return 0;
169
170 // Allocate stack.
171 if((res = impl().initStack(m_stack, m_stackUsable)))
172 return res;
173
174 // Apply guards.
175 impl().valgrindRegister();
176
177 if(unlikely((res = impl().stackGuardInit()))) {
178 // Uh, oh...
179 impl().deallocStack(m_stack);
180 return res;
181 }
182
183 return 0;
184 }

200 int initStack(Stack& stack, Stack& usable) noexcept
201 {
202 stack.size = impl().calcStackSize(stack.size);
203
204 size_t const page = impl().pageSize();
205 zth_assert(__builtin_popcount((unsigned int)page) == 1);
206
207 // Round up to full pages.
208 stack.size = (stack.size + page - 1U) & ~(page - 1U);
209
210 // Add another page size to allow arbitrary allocation location.
211 stack.size += page;
212
213 if(unlikely(!(stack.p = static_cast<char*>(impl().allocStack(stack.size)))))
214 return errno;
215
216 // Compute usable offsets.
217 usable = stack;
218 impl().stackAlign(usable);
219
220 stack_watermark_init(usable.p, usable.size);
221 return 0;
222 }
223

287 void* allocStack(size_t size) noexcept
288 {
289 void* p = nullptr;
290
291# ifdef ZTH_HAVE_MMAN
292 // NOLINTNEXTLINE(hicpp-signed-bitwise,cppcoreguidelines-pro-type-cstyle-cast)
293 p =
294 mmap(nullptr, size, PROT_READ | PROT_WRITE,
295 MAP_PRIVATE | MAP_ANONYMOUS | MAP_STACK, -1, 0);
296
297 if(unlikely(p == MAP_FAILED))
298 return nullptr;
299# else
300 if(unlikely((p = static_cast<void*>(allocate_noexcept<char>(size))) == nullptr)) {
301 errno = ENOMEM;
302 return nullptr;
303 }
304# endif
305
306 return p;
307 }

312 void deallocStack(Stack& stack) noexcept
313 {
314 if(!stack.p)
315 return;
316
317# ifdef ZTH_HAVE_MMAN
318 munmap(stack.p, stack.size);
319# else
320 deallocate(static_cast<char*>(stack.p), stack.size);
321# endif
322
323 stack.p = nullptr;
324 }

329 void stackAlign(Stack& stack) noexcept
330 {
331 if(!stack.p || !stack.size)
332 // Nothing to align.
333 return;
334
335 size_t const ps = impl().pageSize();
336
337 // Align to page size.
338 uintptr_t stack_old = (uintptr_t)stack.p;
339 // calcStackSize() should have taken care of this.
340 zth_assert(stack.size > ps);
341
342 // The stack should not wrap.
343 zth_assert(stack_old + stack.size > stack_old);
344
345 uintptr_t stack_new = ((uintptr_t)(stack_old + ps - 1U) & ~(ps - 1U));
346 stack.size = (stack.size - (stack_new - stack_old)) & ~(ps - 1U);
347
348# ifdef ZTH_HAVE_MMAN
349 if(Config::EnableStackGuard) {
350 // Do not use the pages at both sides.
351 zth_assert(stack.size > ps * 2U);
352 stack_new += ps;
353 stack.size -= ps * 2U;
354 }
355# endif
356
357 zth_assert(stack.size > 0);
358 stack.p = reinterpret_cast<char*>(stack_new); // NOLINT
359 }

393 int stackGuardInit() noexcept
394 {
395# ifdef ZTH_HAVE_MMAN
396 if(Config::EnableStackGuard) {
397 if(!m_stackUsable.p)

401 // Guard both ends of the stack.
402 if(unlikely(mprotect(m_stackUsable.p - ps, ps, PROT_NONE)))
403 return errno;
404 if(unlikely(mprotect(m_stackUsable.p + m_stackUsable.size, ps, PROT_NONE)))
405 return errno;
406 }
