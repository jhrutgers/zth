212  int initStack(Stack& stack, Stack& usable) noexcept
213  {
214  stack.size = impl().calcStackSize(stack.size);
215 
216  size_t const page = impl().pageSize();
217  zth_assert(__builtin_popcount((unsigned int)page) == 1);
218 
219  // Round up to full pages.
220  stack.size = (stack.size + page - 1u) & ~(page - 1u);
221 
222  // Add another page size to allow arbitrary allocation location.
223  stack.size += page;
224 
225  if(unlikely(!(stack.p = (char*)impl().allocStack(stack.size))))
226  return errno;
227 
228  // Compute usable offsets.
229  usable = stack;
230  impl().stackAlign(usable);
231 
232  stack_watermark_init(usable.p, usable.size);
233  return 0;
234  }

341  void stackAlign(Stack& stack) noexcept
342  {
343  if(!stack.p || !stack.size)
344  // Nothing to align.
345  return;
346 
347  size_t const ps = impl().pageSize();
348 
349  // Align to page size.
350  uintptr_t stack_old = (uintptr_t)stack.p;
351  // calcStackSize() should have taken care of this.
352  zth_assert(stack.size > ps);
353 
354  // The stack should not wrap.
355  zth_assert(stack_old + stack.size > stack_old);
356 
357  uintptr_t stack_new = ((uintptr_t)(stack_old + ps - 1u) & ~(ps - 1u));
358  stack.size = (stack.size - (stack_new - stack_old)) & ~(ps - 1u);
359 
360 # ifdef ZTH_HAVE_MMAN
361  if(Config::EnableStackGuard) {
362  // Do not use the pages at both sides.
363  zth_assert(stack.size > ps * 2u);
364  stack_new += ps;
365  stack.size -= ps * 2u;
366  }
367 # endif
368 
369  zth_assert(stack.size > 0);
370  stack.p = (char*)stack_new;
371  }
