131  {
132  zth_assert(elem.prev == nullptr);

134 
135  if(m_tail == nullptr) {
136  zth_assert(m_head == nullptr);
137  m_tail = m_head = elem.prev = elem.next = &elem;

514  {

572  t->left = delete_(x, t->left);

621  {
622  decltype(t->level) ll = t->left ? t->left->level : 0;
