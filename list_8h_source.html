442  elem_type* delete_(elem_type& x, elem_type* t) {

506  static void decrease_level(elem_type* t) {
507  decltype(t->level) ll = t->left ? t->left->level : 0;
508  decltype(t->level) lr = t->right ? t->right->level : 0;
509  decltype(t->level) should_be = (decltype(t->level))((ll < lr ? ll : lr) + 1u);

513  t->right->level = should_be;
