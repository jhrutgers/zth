92 int ZmqPoller::doPoll(int timeout_ms, typename base::PollItemList& items) noexcept
93 {
94  zth_assert(items.size() <= (size_t)std::numeric_limits<int>::max());
95  int res = zmq_poll(&items[0], (int)items.size(), timeout_ms);
96  if(res < 0)
97  return zmq_errno();
98  if(res == 0)
99  // Timeout.
100  return 0;
101 
102  for(size_t i = 0; res > 0 && i < items.size(); i++) {
103  zmq_pollitem_t const& item = items[i];
104  Pollable::Events revents = 0;
105 
106  if(item.revents) {
107  res--;
108 
109  if(item.revents & ZMQ_POLLIN) // NOLINT(hicpp-signed-bitwise)
110  revents |= Pollable::PollIn;
111  if(item.revents & ZMQ_POLLOUT) // NOLINT(hicpp-signed-bitwise)
112  revents |= Pollable::PollOut;
113  if(item.revents & ZMQ_POLLERR) // NOLINT(hicpp-signed-bitwise)
114  revents |= Pollable::PollErr;
115  }
116 
117  this->event(revents, i);
118  }
119 
120  return 0;
121 }

279 PollerClient::Result const& PollerClient::poll(int timeout_ms) noexcept
280 {
281  m_result.clear();
282 
283  if(m_pollables.empty()) {
284  errno = EINVAL;
285  return m_result;
286  }
287 
288  Waiter& waiter = currentWorker().waiter();
289  PollerServerBase& p = waiter.poller();
290 
291  // Add all our pollables to the server.
292  for(size_t i = 0; i < m_pollables.size(); i++)
293  p.add(*m_pollables[i], this);
294 
295  // Go do the poll.
296  zth_dbg(io, "[%s] polling %u items for %d ms", id_str(), (unsigned)m_pollables.size(), timeout_ms);
297 
298  // First try, in the current fiber's context.
299  m_wait = TimedWaitable();
300  int res = p.poll(0);
301 
302  if(!m_result.empty()) { // NOLINT(bugprone-branch-clone)
303  // Completed already.
304  } else if(res && res != EAGAIN) {
305  // Completed with an error.
306  } else if(timeout_ms == 0) {
307  // Just tried. Got nothing, apparently.
308  } else if(timeout_ms < 0) {
309  // Wait for the event callback.
310  zth_dbg(io, "[%s] hand-off to server", id_str());
311  m_wait.setFiber(currentFiber());
312  suspend();
313  } else {
314  zth_dbg(io, "[%s] hand-off to server with timeout", id_str());
315  m_wait = TimedWaitable(Timestamp::now() + TimeInterval(timeout_ms / 1000L, (timeout_ms * 1000000L) % 1000000000L));
316  waitUntil(m_wait);
317  }
318 
319  // Remove our pollables from the server.
320  for(size_t i = m_pollables.size(); i > 0; i--)
321  p.remove(*m_pollables[i - 1u], this);
322 
323  // m_result got populated by event().
324  if(m_result.empty() && !res)
325  res = EAGAIN;
326 
327  if(res)
328  zth_dbg(io, "[%s] poll returned %s", id_str(), err(res).c_str());
329  else
330  zth_dbg(io, "[%s] poll returned %u pollable(s)", id_str(), (unsigned)m_result.size());
331 
332  errno = res;
333  return m_result;
334 }
