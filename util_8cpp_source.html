154void abortv(char const* fmt, va_list args) noexcept
155{
156 static bool recurse = false;
157
158 if(!recurse) {
159 recurse = true;
160 log("\n%s *** Zth ABORT: ", zth::Config::EnableColorLog ? "\x1b[41;1;37;1m" : "");
161 logv(fmt, args);
162 log(" *** %s\n\n", zth::Config::EnableColorLog ? "\x1b[0m" : "");
163
164 Backtrace().print();
165 }
166
167 std::terminate();
168}

191bool log_supports_ansi_colors() noexcept
192{
193#ifdef ZTH_OS_WINDOWS
194 static int support = 0;
195 if(likely(support))
196 return support > 0;
197
198 // Win10 supports ANSI, but it should be enabled. This can be done to
199 // set the Console Mode to ENABLE_VIRTUAL_TERMINAL_PROCESSING (0x0004).
200 // However, the application is not compiled for Win10 specifically, so
201 // we cannot ask if we are running on Win10, and this macro has not
202 // been defined. Therefore, we just try to set the flag by its number,
203 // and check if it succeeds. If so, we assume cmd will accept ANSI
204 // colors.
205 AttachConsole(ATTACH_PARENT_PROCESS);
206
207 DWORD mode = 0;
208 if(!GetConsoleMode(GetStdHandle(STD_OUTPUT_HANDLE), &mode))
209 goto nosupport;
210 if(!SetConsoleMode(GetStdHandle(STD_OUTPUT_HANDLE), mode | 0x0004))
211 goto nosupport;
212
213 // Succeeded. We have support. Enable stderr too.
214 if(GetConsoleMode(GetStdHandle(STD_ERROR_HANDLE), &mode))
215 SetConsoleMode(GetStdHandle(STD_ERROR_HANDLE), mode | 0x0004);
216
217 support = 1;
218 return true;
219
220nosupport:
221 support = -1;
222 return false;
223#else // !ZTH_OS_WINDOWS
224 return true;
225#endif // !ZTH_OS_WINDOWS
226}

233void log_colorv(int color, char const* fmt, va_list args)
234{
235 static bool do_color =
236 Config::EnableColorLog && log_supports_ansi_colors() && isatty(fileno(stdout));
237
238 if(do_color && color > 0)
239 log("\x1b[%d%sm", (color % 8) + 30, color >= 8 ? ";1" : "");
240
241 logv(fmt, args);
242
243 if(do_color && color > 0)
244 zth::log("\x1b[0m");
245}

250#if __cplusplus >= 201703L
251string formatv(char const* fmt, va_list args)
252{
253 // std::string has a small internal buffer. Try to use that first.
254 string res;
255
256# if defined(__cpp_lib_string_resize_and_overwrite) \
257 && __cpp_lib_string_resize_and_overwrite >= 202110L
258 // C++23: Don't initialize the buffer.
259 res.resize_and_overwrite(
260 res.capacity(), [](char* /*buf*/, size_t size) -> size_t { return size; });
261# else // !resize_and_overwrite
262 // This will initialize the buffer with zeros.
263 res.resize(res.capacity());
264# endif // resize_and_overwrite
265
266 va_list args2;
267 va_copy(args2, args);
268 // NOLINTNEXTLINE
269 int c = vsnprintf(res.data(), res.size(), fmt, args2);
270 va_end(args2);
271
272 if(unlikely((size_t)c >= res.size())) {
273 // Buffer too small.
274# if defined(__cpp_lib_string_resize_and_overwrite) \
275 && __cpp_lib_string_resize_and_overwrite >= 202110L
276 res.resize_and_overwrite(
277 (size_t)c + 1U, [](char* /*buf*/, size_t size) -> size_t { return size; });
278# else // !resize_and_overwrite
279 res.resize((size_t)c + 1U);
280# endif // resize_and_overwrite
281
282 // NOLINTNEXTLINE
283 c = vsnprintf(res.data(), res.size(), fmt, args);
284 }
285
286 // Shrink to actual size.
287 res.resize(c > 0 ? (size_t)c : 0);
288 return res; // RVO
289}
290#else // < C++17
291// Pre C++17 version without guaranteed contiguous string storage.
292string formatv(char const* fmt, va_list args)
293{
294 int const maxstack = (int)sizeof(void*) * 8;
295
296 char sbuf[maxstack];
297 char* hbuf = nullptr;
298 size_t hbuf_size = 0;
299 char* buf = sbuf;
300
301 va_list args2;
302 va_copy(args2, args);
303 // NOLINTNEXTLINE
304 int c = vsnprintf(sbuf, maxstack, fmt, args2);
305 va_end(args2);
306
307 if(c >= maxstack) {
308 hbuf_size = (size_t)c + 1U;
309 hbuf = allocate_noexcept<char>(hbuf_size);
310
311 if(unlikely(!hbuf)) {
312 c = 0;
313 } else {
314 // NOLINTNEXTLINE
315 int c2 = vsprintf(hbuf, fmt, args);
316 zth_assert(c2 <= c);
317 c = c2;
318 buf = hbuf;
319 }
320 }
321
322 string res(buf, (size_t)(c < 0 ? 0 : c));
323 deallocate(hbuf, hbuf_size);
324 return res;
325}
326#endif // C++17

352__attribute__((weak)) int vasprintf(char** strp, const char* fmt, va_list ap)
353{
354 va_list ap_dup;
355 va_copy(ap_dup, ap);
356 int len = vsnprintf(nullptr, 0, fmt, ap_dup);
357 va_end(ap_dup);
358
359 if(strp && (*strp = static_cast<char*>(malloc(len + 1))))
360 vsnprintf(*strp, len + 1, fmt, ap);
361
362 // cppcheck-suppress[memleak,unmatchedSuppression]
363 return len;
364}
