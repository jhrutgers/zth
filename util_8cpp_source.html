171 bool log_supports_ansi_colors() noexcept
172 {
173 #ifdef ZTH_OS_WINDOWS
174  static int support = 0;
175  if(likely(support))
176  return support > 0;
177 
178  // Win10 supports ANSI, but it should be enabled.
179  // This can be done to set the Console Mode to ENABLE_VIRTUAL_TERMINAL_PROCESSING (0x0004).
180  // However, the application is not compiled for Win10 specifically, so we cannot ask if we are running on Win10,
181  // and this macro has not been defined.
182  // Therefore, we just try to set the flag by its number, and check if it succeeds.
183  // If so, we assume cmd will accept ANSI colors.
184  AttachConsole(ATTACH_PARENT_PROCESS);
185 
186  DWORD mode = 0;
187  if(!GetConsoleMode(GetStdHandle(STD_OUTPUT_HANDLE), &mode))
188  goto nosupport;
189  if(!SetConsoleMode(GetStdHandle(STD_OUTPUT_HANDLE), mode | 0x0004))
190  goto nosupport;
191 
192  // Succeeded. We have support. Enable stderr too.
193  if(GetConsoleMode(GetStdHandle(STD_ERROR_HANDLE), &mode))
194  SetConsoleMode(GetStdHandle(STD_ERROR_HANDLE), mode | 0x0004);
195 
196  support = 1;
197  return true;
198 
199 nosupport:
200  support = -1;
201  return false;
202 #else
203  return true;
204 #endif
205 }

212 void log_colorv(int color, char const* fmt, va_list args)
213 {
214  static bool do_color =
215  Config::EnableColorLog
216  && log_supports_ansi_colors()
217  && isatty(fileno(stdout));
218 
219  if(do_color && color > 0)
220  log("\x1b[%d%sm", (color % 8) + 30, color >= 8 ? ";1" : "");
221 
222  logv(fmt, args);
223 
224  if(do_color && color > 0)
225  log("\x1b[0m");
226 }

231 std::string formatv(char const* fmt, va_list args)
232 {
233  int const maxstack = (int)sizeof(void*) * 8;
234 
235  char sbuf[maxstack];
236  char* hbuf = nullptr;
237  char* buf = sbuf;
238 
239  va_list args2;
240  va_copy(args2, args);
241  // NOLINTNEXTLINE
242  int c = vsnprintf(sbuf, maxstack, fmt, args);
243 
244  if(c >= maxstack) {
245  hbuf = static_cast<char*>(malloc((size_t)c + 1));
246  if(unlikely(!hbuf)) {
247  c = 0;
248  } else {
249  // NOLINTNEXTLINE
250  int c2 = vsprintf(hbuf, fmt, args2);
251  zth_assert(c2 <= c);
252  c = c2;
253  buf = hbuf;
254  }
255  }
256 
257  va_end(args2);
258 
259  std::string res(buf, (size_t)(c < 0 ? 0 : c));
260  if(hbuf)
261  free(hbuf);
262  return res;
263 }
