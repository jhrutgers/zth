254  else
255  return idle(load(), dt_s);
256  }

259  m_load = idle(load(), dt_s);
260  }
261 
262  void active(type dt_s) {
263  m_load = active(load(), dt_s);
264  }
265 
266  protected:
267  type idle(type load, type dt_s) const {
268  if(dt_s <= 0)
269  return load;
270 
271  return ((type)1 - alpha(dt_s)) * load;
272  }
273 
274  type active(type load, type dt_s) const {
275  if(dt_s <= 0)
276  return load;
277 
278  type a = alpha(dt_s);
279  return a + ((type)1 - a) * load;
280  }
281 
282  type alpha(type dt_s) const {
283  return dt_s / (rc() + dt_s);
284  }
