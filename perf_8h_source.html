280 template <typename... Args>
281 inline void perf_event(Args&&... args) noexcept
282 {
283  if(!Config::EnablePerfEvent || unlikely(!perf_eventBuffer)) {
284  // Drop, as there is no event buffer...
285  return;
286  }
287 
288  try {
289  perf_eventBuffer->emplace_back(std::forward<Args>(args)...);
290  } catch(...) {
291  // Cannot allocate buffer. Drop.
292  return;
293  }
294 
295  zth_assert(perf_eventBuffer->size() <= Config::PerfEventBufferSize);
296 
297  if(unlikely(perf_eventBuffer->size() >= Config::PerfEventBufferThresholdToTriggerVCDWrite))
298  perf_flushEventBuffer();
299 }
300 
301 # define zth_perf_event(...) zth::perf_event(__VA_ARGS__)
302 # else
303 // Copy PerfEvent into buffer.
304 inline void perf_event(PerfEvent<> const& event) noexcept
305 {
306  if(!Config::EnablePerfEvent || unlikely(!perf_eventBuffer)) {
307  // Release now, as there is no event buffer...
308  event.release();
309  return;
310  }
311 
312  try {
313  perf_eventBuffer->push_back(event);
314  } catch(...) {
315  // Cannot allocate buffer. Release now and drop.
316  event.release();
317  return;
318  }
319 
320  zth_assert(perf_eventBuffer->size() <= Config::PerfEventBufferSize);
321 
322  if(unlikely(perf_eventBuffer->size() >= Config::PerfEventBufferThresholdToTriggerVCDWrite))
323  perf_flushEventBuffer();
324 }

435  }

437  type load(Timestamp const& now) const noexcept
438  {
439  type dt_s = (type)(now - m_current).s();
440  if(isActive())
441  return active(load(), dt_s);
442  else
443  return idle(load(), dt_s);
444  }
445 
446  void idle(type dt_s) noexcept
447  {
448  m_load = idle(load(), dt_s);
449  }
450 
451  void active(type dt_s) noexcept
452  {
453  m_load = active(load(), dt_s);
454  }
455 
456 protected:
457  type idle(type load, type dt_s) const noexcept
458  {
459  if(dt_s <= 0)
460  return load;
461 
462  return ((type)1 - alpha(dt_s)) * load;
463  }
464 
465  type active(type load, type dt_s) const noexcept
466  {
467  if(dt_s <= 0)
468  return load;
469 
470  type a = alpha(dt_s);
471  return a + ((type)1 - a) * load;
472  }
473 
474  type alpha(type dt_s) const noexcept
475  {
476  return dt_s / (rc() + dt_s);
477  }
