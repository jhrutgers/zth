#define zth_async

Run a function as a new fiber.

The function must have passed through zth_fiber() (or friends) first. Example:

void foo(int i) { ... }
zth_fiber(foo)
int main_fiber(int argc, char** argv) {
zth_async foo(42);
return 0;
}
